\section*{菱形存在性 · 10题综合练习（教师版）}

\section*{一、入门：单一角色（每题 9 分，共 18 分）}

\needspace{8\baselineskip}

\begin{problem}[points=9]
  在平面直角坐标系中，点 $A(0,0)$、$B(4,0)$。点 $P$ 在 $y$ 轴上运动（异于 $A$），
是否存在点 $P$ 和点 $Q$，使以 $A,B,P,Q$ 为顶点（$AB$ 为一条边）的四边形是菱形？
若存在，求出所有符合条件的点 $P$ 和点 $Q$ 的坐标。

  \begin{answerareawithdiagramtikz}{56mm}{60mm}{第1题图（A、B 与 y 轴）}
  \input{\detokenize{build/diagram/jobs/practice-teacher-q1-prompt/rendered/prompt.fragment.tex}}
  \end{answerareawithdiagramtikz}
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}存在。$P(0,4)$，$Q(4,4)$；或 $P(0,-4)$，$Q(4,-4)$。\par\medskip
    \solstep{设动点}{$AB$ 为边，需 $|AP|=|AB|$。$|AB|=4$。设 $P=(0,t)$（$P$ 在 $y$ 轴，$t\ne0$）。}
    \solstep{列方程}{$|AP|=\sqrt{0^2+t^2}=|t|=4$，所以 $t=4$ 或 $t=-4$。}
    \solstep{补 Q 验四边}{$AB$ 为邻边，顺序 $A\text{-}B\text{-}Q\text{-}P$，$Q=B+P-A$。$t=4$ 时 $P=(0,4)$，$Q=(4,4)$，四边 $AB=BQ=QP=PA=4$；$t=-4$ 时 $P=(0,-4)$，$Q=(4,-4)$，四边均为 $4$。均为菱形。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=9]
  在平面直角坐标系中，点 $A(0,0)$、$B(6,0)$。线段 $AB$ 是菱形 $ABQP$ 的一条对角线，
点 $P$ 在直线 $x=3$ 上，且菱形 $ABQP$ 的面积为 $24$。
求点 $P$ 和点 $Q$ 的坐标，并求菱形的边长。

  \answerarea[56mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$P(3,4)$、$Q(3,-4)$（或互换），边长为 $5$。\par\medskip
    \solstep{用面积求另一条对角线}{面积 $=\dfrac12|AB|\cdot|PQ|=24$，$|AB|=6$，得 $|PQ|=8$。}
    \solstep{定 P,Q 位置}{$AB$ 为对角线、中点 $(3,0)$；$PQ\perp AB$ 且被 $AB$ 平分，$P,Q$ 关于 $x$ 轴对称，横坐标为 $3$。$|PQ|=8$，所以 $P,Q$ 到 $x$ 轴距离为 $4$，得 $P(3,4)$、$Q(3,-4)$（或互换）。}
    \solstep{求边长}{半对角线 $3$ 与 $4$，边长 $=\sqrt{3^2+4^2}=5$。}
  \end{solutionblock}
\end{problem}


\section*{二、完整分类：AB为边 + AB为对角线（每题 10 分，共 20 分）}

\needspace{8\baselineskip}

\begin{problem}[points=10]
  如图，直线 $y=-\dfrac34 x+3$ 与 $x$ 轴交于点 $A$，与 $y$ 轴交于点 $B$。
点 $P$ 在 $x$ 轴上运动（异于点 $A$）。
是否存在点 $P$ 和点 $Q$，使以 $A,B,P,Q$ 为顶点的四边形是菱形？
若存在，求出所有符合条件的点 $P$ 和点 $Q$ 的坐标。

  \begin{answerareawithdiagramtikz}{92mm}{60mm}{第3题图（直线与两坐标轴）}
  \input{\detokenize{build/diagram/jobs/practice-teacher-q3-prompt/rendered/prompt.fragment.tex}}
  \end{answerareawithdiagramtikz}
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}共 3 个 $P$：$(9,0)$、$(-1,0)$、$(\tfrac78,0)$。对应 $Q$ 分别为 $(5,3)$、$(-5,3)$、$(\tfrac{25}{8},3)$。\par\medskip
    \solstep{求 A,B 与 |AB|}{令 $y=0$ 得 $A(4,0)$；令 $x=0$ 得 $B(0,3)$。$|AB|=\sqrt{4^2+3^2}=5$。}
    \solstep{情况一(AB为边)：|AP|=|AB|}{$P=(t,0)$，$|AP|=|t-4|=5$，$t=9$ 或 $t=-1$。$t=9$：$Q=B+P-A=(5,3)$，四边均为 $5$，是菱形；$t=-1$：$Q=(-5,3)$，四边均为 $5$，是菱形。}
    \solstep{情况一(AB为边)：|BP|=|AB|}{$|BP|^2=t^2+9=25$，$t=\pm4$。$t=4$ 即 $A$（退化舍去）；$t=-4$：$P=(-4,0)$，$Q=(-8,3)$，四边 $AB=5$ 但 $BQ=8\ne5$，不是菱形，舍去。}
    \solstep{情况二(AB为对角线)}{$AB$ 中点 $M=(2,\tfrac32)$。$P=(t,0)$，$Q=2M-P=(4-t,3)$。菱形需 $PQ\perp AB$：$\overrightarrow{PQ}=(4-2t,3)$，$\overrightarrow{AB}=(-4,3)$，点积 $8t-7=0$，$t=\tfrac78$。$P=(\tfrac78,0)$，$Q=(\tfrac{25}{8},3)$，四边平方均为 $\tfrac{25}{8}$，是菱形。}
    \solstep{结论}{共 3 个 $P$：$(9,0)$、$(-1,0)$、$(\tfrac78,0)$，对应 $Q$ 为 $(5,3)$、$(-5,3)$、$(\tfrac{25}{8},3)$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  直线 $y=-\dfrac34 x+6$ 与 $x$ 轴交于点 $A$，与 $y$ 轴交于点 $B$。
点 $P$ 在 $x$ 轴上运动（异于点 $A$）。
是否存在点 $P$ 和点 $Q$，使以 $A,B,P,Q$ 为顶点的四边形是菱形？
若存在，求出所有符合条件的点 $P$ 和点 $Q$ 的坐标。

  \answerarea[92mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}共 3 个 $P$：$(18,0)$、$(-2,0)$、$(\tfrac74,0)$。对应 $Q$ 分别为 $(10,6)$、$(-10,6)$、$(\tfrac{25}{4},6)$。\par\medskip
    \solstep{求 A,B 与 |AB|}{令 $y=0$ 得 $A(8,0)$；令 $x=0$ 得 $B(0,6)$。$|AB|=\sqrt{8^2+6^2}=10$。}
    \solstep{情况一(AB为边)：|AP|=|AB|}{$|AP|=|t-8|=10$，$t=18$ 或 $t=-2$。$t=18$：$Q=B+P-A=(10,6)$，四边均为 $10$，是菱形；$t=-2$：$Q=(-10,6)$，四边均为 $10$，是菱形。}
    \solstep{情况一(AB为边)：|BP|=|AB|}{$|BP|^2=t^2+36=100$，$t=\pm8$。$t=8$ 即 $A$（退化舍去）；$t=-8$：$P=(-8,0)$，$Q=(-16,6)$，$BQ^2=256+36\ne100$，不是菱形，舍去。}
    \solstep{情况二(AB为对角线)}{$AB$ 中点 $M=(4,3)$。$P=(t,0)$，$Q=2M-P=(8-t,6)$。$PQ\perp AB$：$\overrightarrow{PQ}=(8-2t,6)$，$\overrightarrow{AB}=(-8,6)$，点积 $-8(8-2t)+36=0$，$16t=28$，$t=\tfrac74$。$P=(\tfrac74,0)$，$Q=(\tfrac{25}{4},6)$，四边平方均为 $\tfrac{625}{16}$，是菱形。}
    \solstep{结论}{共 3 个 $P$：$(18,0)$、$(-2,0)$、$(\tfrac74,0)$，对应 $Q$ 为 $(10,6)$、$(-10,6)$、$(\tfrac{25}{4},6)$。}
  \end{solutionblock}
\end{problem}


\section*{三、轨迹与角色：天然成立 / 明确无解（每题 10 分，共 20 分）}

\needspace{8\baselineskip}

\begin{problem}[points=10]
  在平面直角坐标系中，点 $A(0,0)$、$B(8,0)$。线段 $AB$ 是菱形 $ABQP$ 的一条对角线，
点 $P$ 在直线 $x=4$（即 $AB$ 的垂直平分线）上，菱形 $ABQP$ 的面积为 $24$。
求点 $P$ 和点 $Q$ 的坐标，并求菱形的边长。

  \answerarea[62mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$P(4,3)$、$Q(4,-3)$（或互换），边长为 $5$。\par\medskip
    \solstep{用面积求 |PQ|}{面积 $=\dfrac12|AB|\cdot|PQ|=24$，$|AB|=8$，得 $|PQ|=6$。}
    \solstep{定 P,Q}{$P$ 在中垂线 $x=4$ 上，$P=(4,t)$，$Q$ 关于 $x$ 轴对称为 $(4,-t)$，$|PQ|=2|t|=6$，$|t|=3$。故 $P(4,3)$、$Q(4,-3)$（或互换）。}
    \solstep{求边长}{半对角线 $4$ 与 $3$，边长 $=\sqrt{4^2+3^2}=5$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  在平面直角坐标系中，点 $A(0,0)$、$B(6,0)$。点 $P$ 在 $y$ 轴上运动（异于 $A$）。
是否存在点 $P$ 和点 $Q$，使以 $A,B,P,Q$ 为顶点的四边形是菱形？
若存在，求出所有符合条件的点 $P$ 和点 $Q$ 的坐标；若某一类不存在，请说明理由。

  \answerarea[78mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}存在，$P(0,6)$、$Q(6,6)$ 或 $P(0,-6)$、$Q(6,-6)$。（“$AB$ 为对角线”一类无解）\par\medskip
    \solstep{情况一(AB为边)}{$|AB|=6$。$P=(0,t)$，$|AP|=|t|=6$，$t=\pm6$。$t=6$：$Q=B+P-A=(6,6)$，四边均为 $6$，是菱形；$t=-6$：$Q=(6,-6)$，四边均为 $6$，是菱形。}
    \solstep{情况一(AB为边)：|BP|=|AB|}{$|BP|^2=36+t^2=36$，$t=0$ 即 $A$（退化舍去），无新增解。}
    \solstep{情况二(AB为对角线)}{$AB$ 水平，若 $AB$ 为对角线则 $PQ\perp AB$，$PQ$ 必须竖直，即 $P,Q$ 横坐标相同。但 $P$ 在 $y$ 轴（$x=0$），$Q$ 关于 $AB$ 中点 $(3,0)$ 对称得 $Q_x=6\ne0$，矛盾。故此类无解。}
    \solstep{结论}{只有“$AB$ 为边”一类有解：$P(0,6)$、$Q(6,6)$ 或 $P(0,-6)$、$Q(6,-6)$。}
  \end{solutionblock}
\end{problem}


\section*{四、迁移与筛选（每题 10 分，共 20 分）}

\needspace{8\baselineskip}

\begin{problem}[points=10]
  直线 $y=-\dfrac43 x+4$ 与 $x$ 轴交于点 $A$，与 $y$ 轴交于点 $B$。
点 $P$ 在 $x$ 轴上运动（异于点 $A$）。
是否存在点 $P$ 和点 $Q$，使以 $A,B,P,Q$ 为顶点的四边形是菱形？
若存在，求出所有符合条件的点 $P$ 和点 $Q$ 的坐标。

  \answerarea[92mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}共 3 个 $P$：$(8,0)$、$(-2,0)$、$(-\tfrac76,0)$。对应 $Q$ 分别为 $(5,4)$、$(-5,4)$、$(\tfrac{25}{6},4)$。\par\medskip
    \solstep{求 A,B 与 |AB|}{令 $y=0$ 得 $A(3,0)$；令 $x=0$ 得 $B(0,4)$。$|AB|=\sqrt{3^2+4^2}=5$。}
    \solstep{情况一(AB为边)：|AP|=|AB|}{$|AP|=|t-3|=5$，$t=8$ 或 $t=-2$。$t=8$：$Q=B+P-A=(5,4)$，四边均为 $5$，是菱形；$t=-2$：$Q=(-5,4)$，四边均为 $5$，是菱形。}
    \solstep{情况一(AB为边)：|BP|=|AB|}{$|BP|^2=t^2+16=25$，$t=\pm3$。$t=3$ 即 $A$（退化舍去）；$t=-3$：$P=(-3,0)$，$Q=(-6,4)$，$BQ^2=36+16\ne25$，不是菱形，舍去。}
    \solstep{情况二(AB为对角线)}{$AB$ 中点 $M=(\tfrac32,2)$。$P=(t,0)$，$Q=2M-P=(3-t,4)$。$PQ\perp AB$：$\overrightarrow{PQ}=(3-2t,4)$，$\overrightarrow{AB}=(-3,4)$，点积 $-3(3-2t)+16=0$，$6t=-7$，$t=-\tfrac76$。$P=(-\tfrac76,0)$，$Q=(\tfrac{25}{6},4)$，四边平方均为 $\tfrac{625}{36}$，是菱形。}
    \solstep{结论}{共 3 个 $P$：$(8,0)$、$(-2,0)$、$(-\tfrac76,0)$，对应 $Q$ 为 $(5,4)$、$(-5,4)$、$(\tfrac{25}{6},4)$。}
  \end{solutionblock}
\end{problem}

\begin{problem}[points=10]

点 $P$ 在 $y$ 轴上运动（异于点 $B$）。
是否存在点 $P$ 和点 $Q$，使以 $A,B,P,Q$ 为顶点的四边形是菱形？

\section*{五、反向构造（每题 11 分，共 22 分）}

\needspace{8\baselineskip}

\begin{problem}[points=11]
  已知一个菱形的边长为 $13$，一条对角线长为 $10$。
\begin{enumerate}[label=(\arabic*)]
\ \  \item 求这个菱形另一条对角线的长；
  \item 求这个菱形的面积；
  \item 把这个菱形放入平面直角坐标系，使长为 $10$\ \ 的对角线与 $x$ 轴重合、
        两条对角线的交点在原点，写出四个顶点的坐标。
\end{enumerate}

  \answerarea[82mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}(1) 另一对角线长 $24$；(2) 面积 $120$；(3) $(-5,0),(5,0),(0,12),(0,-12)$。\par\medskip
    \solstep{求另一条对角线}{边长 $13$，一对角线 $10$，半对角线 $5$。设另一半对角线为 $x$，$5^2+x^2=13^2$，$x^2=144$，$x=12$。另一对角线长 $=2x=24$。}
    \solstep{求面积}{面积 $=\dfrac12\times10\times24=120$。}
    \solstep{放坐标系写顶点}{长为 $10$ 的对角线在 $x$ 轴、交点在原点，两端点 $(-5,0),(5,0)$；另一对角线在 $y$ 轴，两端点 $(0,12),(0,-12)$。}
    \solstep{验证}{相邻两顶点 $(5,0)$ 与 $(0,12)$ 距离平方 $=25+144=169=13^2$，边长正确。}
  \end{solutionblock}
\end{problem}

\begin{problem}[points=11]
  在平面直角坐标系中，点 $A(0,0)$、$B(10,0)$。线段 $AB$ 是菱形 $ABQP$ 的一条对角线，
点 $P$ 在直线 $x=5$ 上，且菱形的边长为 $\sqrt{34}$。
求点 $P$ 和点 $Q$ 的坐标。

  \answerarea[62mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$P(5,3)$、$Q(5,-3)$ 或 $P(5,-3)$、$Q(5,3)$。\par\medskip
    \solstep{设 P 并列方程}{$AB$ 为对角线，$P$ 在 $x=5$ 上，$P=(5,t)$。$AB$ 中点 $(5,0)$，$Q=(5,-t)$。边长平方 $=|AP|^2=5^2+t^2=25+t^2$。}
    \solstep{解 t}{边长为 $\sqrt{34}$，$25+t^2=34$，$t^2=9$，$t=\pm3$。}
    \solstep{写 P,Q}{$t=3$：$P(5,3)$、$Q(5,-3)$；$t=-3$：$P(5,-3)$、$Q(5,3)$。两组等价（互换）。}
  \end{solutionblock}
\end{problem}
