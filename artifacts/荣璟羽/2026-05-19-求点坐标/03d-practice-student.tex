\section*{通过交点、面积、距离等条件求点坐标（四） · 练习卷}

\section*{一、选择题（每题 12 分，共 48 分）}


\begin{keyideabox}
  \textbf{中等思想：坐标面积公式}
  已知 $A(x_a, y_a)$，$B(x_b, y_b)$，$C(x_c, y_c)$，三角形面积为：\\ $S = \frac{1}{2}|x_a(y_b - y_c) + x_b(y_c - y_a) + x_c(y_a - y_b)|$。这能避免复杂的割补画图。
\end{keyideabox}



\begin{question}[points=12]
  已知点 $O(0, 0)$，点 $A(0, 4)$。若点 $P$ 在直线 $y = -2x$ 上，且 $\triangle OAP$ 的面积为 $8$，则点 $P$ 的坐标为
  \begin{choices}
    \item $(4, -8)$ 或 $(-4, 8)$
    \item $(4, -8)$
    \item $(2, -4)$ 或 $(-2, 4)$
    \item $(2, -4)$
  \end{choices}
\end{question}



\begin{question}[points=12]
  直线 $y = -x + 4$ 与 $x$ 轴交于点 $A$，与 $y$ 轴交于点 $B$。若点 $C$ 在 $y$ 轴上，且 $\triangle ABC$ 的面积为 2，则点 $C$ 的坐标可以为
  \begin{choices}
    \item $(0, 5)$ 或 $(0, 3)$
    \item $(0, 5)$
    \item $(0, 3)$
    \item $(0, 6)$
  \end{choices}
\end{question}



\begin{question}[points=12]
  已知三点 $A(1, 0)$，$B(0, 3)$，$C(2, 4)$，则 $\triangle ABC$ 的面积为
  \begin{choices}
    \item 3.5
    \item 4.5
    \item 5.5
    \item 2.5
  \end{choices}
\end{question}



\begin{question}[points=12]
  已知直线平行于 $y = -3x$，且经过点 $C(1, -2)$，则该直线与 $x$ 轴的交点横坐标为
  \begin{choices}
    \item 1/3
    \item -1/3
    \item 3
    \item -3
  \end{choices}
\end{question}


\section*{二、填空题（每题 12 分，共 12 分）}


\begin{question}[points=12]
  已知 $A(-2, 0)$，$C(4, 6)$。点 $P$ 是 $y$ 轴上的一个动点，若要使线段长度之差 $|PA - PC|$ 达到最大，则此时点 $P$ 的坐标为 \fillin。
\end{question}

\begin{problem}[points=20]
  直线 $y = x + 3$ 与 $x$ 轴交于点 $A$，与 $y$ 轴交于点 $B$。另有一条过原点 $O$ 的直线 $l$ 与线段 $AB$ 交于点 $C$。

(1) 求点 $A$ 和点 $B$ 的坐标，并计算 $\triangle AOB$ 的面积；
(2) 若直线 $l$ 将 $\triangle AOB$ 的面积分为 $2:1$ 两部分，求直线 $l$ 的解析式。
  \answerarea[60mm]
\end{problem}

(1) 当四边形 $BDEC$ 的周长最小时，请利用平移和对称画出点 $D$ 和点 $E$ 的位置并简述理由；
