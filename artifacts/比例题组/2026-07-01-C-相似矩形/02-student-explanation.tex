\eduSectionTitle{相似矩形：先定对应，再列比例}


\begin{eduSolutionBox}{先定对应，再选择方法}
  \begingroup
  \setlength{\parskip}{2.5mm}
判断相似可以比较各自长宽比：甲长:甲宽 是否等于 乙长:乙宽。\par
也可以比较对应边倍率：甲长:乙长 是否等于 甲宽:乙宽。\par
求未知边先定对应——题目说"长、宽"时通常长对长、宽对宽。\par
列出比例后沿用比例方程的方法选择：先看明显倍数，再看份数，最后才交叉相乘。\par
只说"两边长为 a 和 b"而未指定对应 $\to$ 要分类讨论（两种对应两种答案）。\par
干扰项：图形旋转后对应混乱；没说 $6$ 对应哪条边时漏掉一种情况。\par
  \endgroup
\end{eduSolutionBox}

\vspace{0pt}
矩形甲的长为 $6$、宽为 $4$，矩形乙的长为 $12$、宽为 $8$。判断两个矩形是否相似。
\end{minipage}\hfill

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
识别长宽比$\;\to\;$比较判断$\;\to\;$结论\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 1}
判断两个矩形是否相似。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{看·转·防}
        \eduSideHint{看结构}{判断题先写两个长宽比，不先列未知数。}
        \eduSideHint{怎么转}{长对长、宽对宽，分别化简后比较。}
        \eduSideMistake{易错}{不要只看数字像不像，要算出化简比。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{识别长宽比}{甲长宽比 $6:4=3:2$，乙长宽比 $12:8=3:2$。}
      \eduExplainStep{2}{比较判断}{两个长宽比相同，故相似。}
      \eduExplainStep{3}{结论}{两个矩形相似。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduProblemBlock}[例题 2]
\noindent

\begin{eduAuxItem}{思路导航}
识别长宽比（含分数）$\;\to\;$比较判断$\;\to\;$结论\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 2}
判断两个矩形是否相似。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{看·转·防}
        \eduSideHint{看结构}{分数边长照样写成长宽比，先约去分母。}
        \eduSideHint{怎么转}{$\dfrac{3}{2}:\dfrac{5}{2}$ 同乘 $2$ 得 $3:5$。}
        \eduSideMistake{易错}{不能因为数字差不多就判相似；$3:5$ 与 $4:7$ 实际不等。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{识别长宽比（含分数）}{甲 $\dfrac{3}{2}:\dfrac{5}{2}=3:5$，乙 $4:7$。}
      \eduExplainStep{2}{比较判断}{$3:5\ne 4:7$，两个长宽比不同。}
      \eduExplainStep{3}{结论}{两个矩形不相似。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduProblemBlock}[例题 3]
\noindent
\begin{minipage}[t]{\dimexpr\linewidth-56mm-6mm\relax}
\vspace{0pt}
两个矩形相似。矩形甲的长为 $8$、宽为 $x+1$；矩形乙的长为 $4$、宽为 $3$。求 $x$。
\end{minipage}\hfill
\begin{diagramcoltikz}{56mm}{长对长、宽对宽，列比例后先选方法。}
\begin{tikzpicture}[x=0.8cm,y=0.8cm,baseline=(current bounding box.center)]
  \draw[line width=1pt] (0,0) rectangle (4.0,1.6);
  \node at (2.0,-0.4) {\tiny 甲 长 $8$，宽 $x+1$};
  \draw[line width=1pt] (5.2,0) rectangle (6.8,1.2);
  \node at (6.0,-0.4) {\tiny 乙 $4\times 3$};
\end{tikzpicture}
\end{diagramcoltikz}
\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
定对应$\;\to\;$选方法$\;\to\;$计算\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 3}
求 $x$。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{看·转·防}
        \eduSideHint{看结构}{已知相似求边，先定对应——长对长、宽对宽。}
        \eduSideHint{怎么转}{列出比例后先选方法；本题有明显倍数，优先用倍数法。}
        \eduSideMistake{易错}{$x+1$ 是整体；若没有明显倍数再交叉相乘，也要把括号保留。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{定对应}{相似 $\Rightarrow$ 长对长、宽对宽：$\dfrac{8}{4}=\dfrac{x+1}{3}$。}
      \eduExplainStep{2}{选方法}{先看明显倍数，$4\to 8$ 是 $\times 2$，所以 $3\to x+1$ 也应 $\times 2$。}
      \eduExplainStep{3}{计算}{$x+1=6$，所以 $x=5$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduProblemBlock}[例题 4]

\vspace{0pt}
一个矩形的两边长为 $2$ 和 $3$。另一个矩形的一边长为 $6$，另一边长为 $x$。若两个矩形相似，求 $x$。
\end{minipage}\hfill

\end{diagramcoltikz}
\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
识别对应是否唯一$\;\to\;$情况一（6 对应 2）$\;\to\;$情况二（6 对应 3）$\;\to\;$合并答案\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 4}
求 $x$。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{看·转·防}
        \eduSideMistake{看结构（核心）}{只说两边长为 $2$ 和 $3$、没说 $6$ 对应谁 $\to$ 分类信号，不能只列一种。}
        \eduSideHint{怎么转}{分别假设 $6$ 对应 $2$、$6$ 对应 $3$，各列一个比例。}
        \eduSideMistake{易错}{漏掉第二种对应是最常见错误；答案要用"或"连接。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{识别对应是否唯一}{题目没说 $6$ 对应 $2$ 还是 $3$ $\Rightarrow$ 对应不唯一，要分类。}
      \eduExplainStep{2}{情况一（6 对应 2）}{$2:3=6:x \Rightarrow x=9$。}
      \eduExplainStep{3}{情况二（6 对应 3）}{$2:3=x:6 \Rightarrow x=4$。}
      \eduExplainStep{4}{合并答案}{所以 $x=9$ 或 $x=4$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduMistakeBox}
\eduSubTitle{易错提醒}相似矩形题最容易错在对应关系没判定就计算。判断题先写长宽比；求边题先写谁对应谁；未指定对应先分类。\end{eduMistakeBox}




\begin{eduSummaryBlock}{\eduIconPin}{方法提醒}
\begin{eduBulletList}
  \item 判断题——先写两个长宽比，化简后比较。  \item 求边题——先写对应边关系（长对长、宽对宽），再选倍数、份数或交叉相乘。  \item 未指定对应——分两种情况，答案用"或"连接。\end{eduBulletList}
\end{eduSummaryBlock}
