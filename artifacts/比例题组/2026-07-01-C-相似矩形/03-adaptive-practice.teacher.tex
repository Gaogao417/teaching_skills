\needspace{8\baselineskip}

\begin{problem}[points=6]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    矩形甲的长为 $8$、宽为 $5$，矩形乙的长为 $16$、宽为 $10$。判断两个矩形是否相似。
  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{矩形示意，边长以题干为准。}
  \input{\detokenize{build/diagram/jobs/cq1-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[28mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}相似\par\medskip
    \solstep{识别长宽比}{甲 $8:5$，乙 $16:10=8:5$。}
    \solstep{比较判断}{两个长宽比相同，故相似。}
  \end{solutionblock}
\end{problem}

\begin{problem}[points=6]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    一个矩形的两边长为 $3$ 和 $5$。另一个矩形的一边长为 $15$，另一边长为 $x$。若两个矩形相似，求 $x$。
  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{15 可能对应 3，也可能对应 5，要分两种情况。}
  \input{\detokenize{build/diagram/jobs/cq7-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[28mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$x=25$ 或 $x=9$\par\medskip
    \solstep{识别对应是否唯一}{没说 $15$ 对应 $3$ 还是 $5$，对应不唯一，要分类。}
    \solstep{情况一（15 对应 3）}{$3:5=15:x$，选倍数法，$3\to15$ 是 $\times5$，所以 $x=25$。}
    \solstep{情况二（15 对应 5）}{$3:5=x:15$，选倍数法，$5\to15$ 是 $\times3$，所以 $x=9$。}
  \end{solutionblock}
\end{problem}
