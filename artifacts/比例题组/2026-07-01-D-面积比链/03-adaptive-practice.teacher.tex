\section*{第1组：基础面积转换}

\needspace{8\baselineskip}

\begin{problem}[points=8]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    在 $\triangle ABC$ 中，点 $E$ 在边 $AC$ 上，点 $O$ 不在直线 $AC$ 上。若 $AE:EC=2:3$，且 $S_{\triangle OAE}=6$，求 $S_{\triangle OEC}$。
  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{根据题干观察点的位置关系。}
  \input{\detokenize{build/diagram/jobs/dq1-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[42mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$9$\par\medskip
    \solstep{同高转底}{$\triangle OAE$ 与 $\triangle OEC$ 同高，面积比为 $2:3$。$6$ 对应 $2$ 份，所以 $S_{\triangle OEC}=9$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=8]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    在 $\triangle ABE$ 中，点 $O$ 在边 $BE$ 上。若 $BO:OE=5:2$，且 $S_{\triangle OAE}=4$，求 $S_{\triangle OAB}$。
  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{根据题干观察点的位置关系。}
  \input{\detokenize{build/diagram/jobs/dq2-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[42mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$10$\par\medskip
    \solstep{共底转高}{$\triangle OAB$ 与 $\triangle OAE$ 共底 $AO$，面积比为 $BO:OE=5:2$，所以 $S_{\triangle OAB}=10$。}
  \end{solutionblock}
\end{problem}


\section*{第2组：四边形拆分综合}

\needspace{8\baselineskip}

\begin{problem}[points=8]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    在 $\triangle ABC$ 中，点 $D$ 在边 $BC$ 上，点 $E$ 在边 $AC$ 上，线段 $AD$ 与 $BE$ 交于点 $O$。若 $AO:OD=1:1$，$BO:OE=3:1$，$AE:EC=1:2$，求 $S_{OECD}:S_{\triangle OAB}$。
  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{根据题干观察点的位置关系。}
  \input{\detokenize{build/diagram/jobs/dq3-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[42mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$5:3$\par\medskip
    \solstep{拆四边形}{$S_{OECD}=S_{\triangle OEC}+S_{\triangle OCD}$。}
    \solstep{统一份数}{设 $S_{\triangle OAE}=1$，则 $S_{\triangle OAB}=3$，$S_{\triangle OEC}=2$，$S_{\triangle OAC}=3$，$S_{\triangle OCD}=3$。所以目标比为 $5:3$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=8]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    在 $\triangle ABC$ 中，点 $D$ 在边 $BC$ 上，点 $E$ 在边 $AC$ 上，线段 $AD$ 与 $BE$ 交于点 $O$。若 $AO:OD=1:1$，$BO:OE=2:1$，$AE:EC=1:3$，求 $S_{OECD}:S_{\triangle OAB}$。
  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{根据题干观察点的位置关系。}
  \input{\detokenize{build/diagram/jobs/dq4-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[42mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$7:2$\par\medskip
    \solstep{统一份数}{设 $S_{\triangle OAE}=1$，则 $S_{\triangle OAB}=2$，$S_{\triangle OEC}=3$，$S_{\triangle OAC}=4$。由 $AO:OD=1:1$ 得 $S_{\triangle OCD}=4$，所以 $S_{OECD}=7$，目标比为 $7:2$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=8]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    在 $\triangle ABC$ 中，点 $D$ 在边 $BC$ 上，点 $E$ 在边 $AC$ 上，线段 $AD$ 与 $BE$ 交于点 $O$。若 $BO:OE=2:1$，$AE:EC=1:2$，且 $S_{OECD}:S_{\triangle OAB}=11:5$，求 $AO:OD$。
  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{根据题干观察点的位置关系。}
  \input{\detokenize{build/diagram/jobs/dq5-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[42mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$5:4$\par\medskip
    \solstep{反推}{为避免分数，设 $S_{\triangle OAE}=5$，则 $S_{\triangle OAB}=10$，$S_{\triangle OEC}=10$，$S_{\triangle OAC}=15$。由 $S_{OECD}:S_{\triangle OAB}=11:5$ 得 $S_{OECD}=22$，故 $S_{\triangle OCD}=12$，所以 $AO:OD=15:12=5:4$。}
  \end{solutionblock}
\end{problem}
