\begin{eduProblemBlock}[题目]
在 $\triangle ABC$ 中，$\angle C = 90^\circ$，$AC = 3$，$BC = 4$。
\begin{enumerate}[label=(\arabic*)]
  \item 若以点 $C$ 为圆心作 $\odot C$，使 $\odot C$ 与斜边 $AB$ 相切，求 $\odot C$ 的半径 $r$；
  \item 若以点 $C$ 为圆心，以 $2.5$ 为半径作 $\odot C$，判断直线 $AB$ 与 $\odot C$ 的位置关系。
\end{enumerate}

\end{eduProblemBlock}









\begin{eduAuxItem}{思路导航}
利用勾股定理求斜边 $AB$ 的长度$\;\to\;$利用等面积法求圆心 $C$ 到斜边 $AB$ 的距离 $d$$\;\to\;$第（1）问：由相切条件 $d = r$ 确定半径$\;\to\;$第（2）问：比较距离 $d$ 与半径 $r = 2.5$ 的大小确定位置关系\end{eduAuxItem}




\begin{eduAuxItem}{核心思路}
圆心到直线的距离 $d$ 即点到直线的垂线段长度。
直角三角形斜边上的高常用等面积法求解：
$$d = \frac{a \cdot b}{c}$$
确定 $d$ 后，通过比较 $d$ 与 $r$ 的大小判定直线与圆的位置关系。
\end{eduAuxItem}




\begin{eduSubQuestionItem}{(1)}
若以点 $C$ 为圆心作 $\odot C$，使 $\odot C$ 与斜边 $AB$ 相切，求 $\odot C$ 的半径 $r$。\end{eduSubQuestionItem}

\begin{eduSolutionBox}{解答}
  \begingroup
  \setlength{\parskip}{2.5mm}
在 $Rt\triangle ABC$ 中，由勾股定理得：$AB = \sqrt{AC^2 + BC^2} = \sqrt{3^2 + 4^2} = 5$。\par
设圆心 $C$ 到斜边 $AB$ 的距离为 $d$。由等面积法得：$AC \cdot BC = AB \cdot d$。\par
代入数据：$3 \times 4 = 5 \times d \implies d = 2.4$。\par
因 $\odot C$ 与斜边 $AB$ 相切，故半径 $r = d = 2.4$。\par
  \endgroup
  \vspace{2.5mm}
  {\color{edu-blue}\bfseries\small 注意}\par\vspace{1.5mm}
  \begin{eduBulletList}
    \item 圆心到斜边 $AB$ 的垂直距离 $d = 2.4$ 是由三角形确定的几何特征值，不随圆的半径变化而改变，在第（2）问中可直接复用。  \end{eduBulletList}
\end{eduSolutionBox}




\begin{eduSubQuestionItem}{(2)}
若以点 $C$ 为圆心，以 $2.5$ 为半径作 $\odot C$，判断直线 $AB$ 与 $\odot C$ 的位置关系。\end{eduSubQuestionItem}

\begin{eduSolutionBox}{解答}
  \begingroup
  \setlength{\parskip}{2.5mm}
由第（1）问知，圆心 $C$ 到斜边 $AB$ 的距离 $d = 2.4$。\par
已知半径 $r = 2.5$。\par
因 $d = 2.4 < 2.5 = r$，即 $d < r$。\par
故直线 $AB$ 与 $\odot C$ 相交。\par
  \endgroup
\end{eduSolutionBox}

\begin{eduMistakeBox}
\eduSubTitle{位置关系名词写错}回答位置关系时，须使用标准的几何名词：\textbf{相交、相切、相离}。避免使用“相割”或“有两个公共点”等非标准表述，以免失分。
\end{eduMistakeBox}




\begin{eduMistakeBox}
\eduSubTitle{误以为 d 会随着 r 的变化而变化}误以为 $d$ 随 $r$ 的变化而变化。
事实上，只要三角形的大小和圆心的位置固定，圆心到斜边的距离 $d$ 即为定值，在同一题中保持恒定，与圆的半径变化无关。
\end{eduMistakeBox}
