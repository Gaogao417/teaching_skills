\eduSectionTitle{知识点 1：配方法把一般式化为顶点式}


\begin{eduSolutionBox}{核心结论：配方法三步——提 a、配完全平方、合并}
  \begingroup
  \setlength{\parskip}{2.5mm}
目标：把 $y=ax^2+bx+c$（$a\ne0$）化为 $y=a(x-h)^2+k$。\par
\textbf{第一步（提 $a$）：} 只把 $x^2$ 和 $x$ 项提出 $a$，常数 $c$ 留在括号外：$y=a\!\left(x^2+\dfrac{b}{a}x\right)+c$。\par
\textbf{第二步（配完全平方）：} 括号内加减 $\left(\dfrac{b}{2a}\right)^2$：$x^2+\dfrac{b}{a}x=\left(x+\dfrac{b}{2a}\right)^2-\left(\dfrac{b}{2a}\right)^2$。\par
\textbf{第三步（合并）：} 把括号内减掉的常数乘 $a$ 后与 $c$ 合并得 $k$，写出 $a(x-h)^2+k$，其中 $h=-\dfrac{b}{2a}$，$k=\dfrac{4ac-b^2}{4a}$。\par
  \endgroup
\end{eduSolutionBox}




\begin{eduProblemBlock}[例题 1（a=1）]
用配方法把二次函数 $y=x^2-4x+1$ 化为顶点式 $y=a(x-h)^2+k$，并写出顶点。

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
提 $a=1$（$c$ 留外）$\;\to\;$括号内配完全平方$\;\to\;$合并得顶点式\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 1}
把 $y=x^2-4x+1$ 化为顶点式。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{配什么数}{一次项系数是 $-4$，它的一半是 $-2$，平方是 $4$；所以加减 $4$。}
        \eduSideMistake{加减要平衡}{加了 $4$ 就要在括号内减 $4$，否则等式被破坏。}
        \eduSideHint{怎么验}{用 $h=-\dfrac{b}{2a}=-\dfrac{-4}{2}=2$ 反验，应得到同样的 $h=2$。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{提 $a=1$（$c$ 留外）}{$y=(x^2-4x)+1$。}
      \eduExplainStep{2}{括号内配完全平方}{$x^2-4x=(x^2-4x+4)-4=(x-2)^2-4$。}
      \eduExplainStep{3}{合并得顶点式}{$y=(x-2)^2-4+1=(x-2)^2-3$，顶点为 $(2,-3)$。}
    \vspace{1mm}
    {\color{edu-blue}\bfseries\small 验算}\par\vspace{1mm}
    \begin{eduBulletList}
      \item 展开 $(x-2)^2-3=x^2-4x+4-3=x^2-4x+1$，与原式一致。    \end{eduBulletList}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduProblemBlock}[例题 2（a≠1，训练提 a 与乘回符号）]
用配方法把二次函数 $y=2x^2-8x+5$ 化为顶点式 $y=a(x-h)^2+k$。

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
提 $a=2$（$c=5$ 留外）$\;\to\;$括号内配完全平方$\;\to\;$乘回后合并得顶点式\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 2}
把 $y=2x^2-8x+5$ 化为顶点式。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideMistake{c 不要除以 a}{提 $a=2$ 时只提 $2x^2-8x$，常数 $5$ 留在括号外面，不要写成 $2(x^2-4x+2.5)$。}
        \eduSideMistake{减回去的常数要乘 a}{括号内减了 $4$，乘 $a=2$ 后是 $-8$；不要漏乘只写 $-4$。}
        \eduSideHint{公式反验}{$h=-\dfrac{-8}{2\cdot2}=2$，$k=\dfrac{4\cdot2\cdot5-(-8)^2}{4\cdot2}=\dfrac{40-64}{8}=-3$，与配方结果一致。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{提 $a=2$（$c=5$ 留外）}{$y=2(x^2-4x)+5$。}
      \eduExplainStep{2}{括号内配完全平方}{$x^2-4x=(x-2)^2-4$，代入：$y=2[(x-2)^2-4]+5$。}
      \eduExplainStep{3}{乘回后合并得顶点式}{$y=2(x-2)^2-8+5=2(x-2)^2-3$。}
    \vspace{1mm}
    {\color{edu-blue}\bfseries\small 验算}\par\vspace{1mm}
    \begin{eduBulletList}
      \item 展开 $2(x-2)^2-3=2(x^2-4x+4)-3=2x^2-8x+8-3=2x^2-8x+5$，与原式一致。    \end{eduBulletList}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduMistakeBox}
\eduSubTitle{易错提醒}\begin{itemize}
  \item \textbf{提 $a$ 时把 $c$ 也除以 $a$}：最常见错误。$c$ 要留在括号外。
  \item \textbf{减回去的常数漏乘 $a$}：括号内减的 $\left(\frac{b}{2a}\right)^2$ 要乘 $a$ 后再合并。
  \item \textbf{$h$ 符号反}：顶点式是 $x-h$，所以 $h=-\frac{b}{2a}$，括号里 $x+\frac{b}{2a}$ 对应 $h=-\frac{b}{2a}$。
  \item \textbf{配完全平方不加不减}：加了 $\left(\frac{b}{2a}\right)^2$ 必须在括号内再减掉。
\end{itemize}
\end{eduMistakeBox}
