\eduSectionTitle{知识点 6：二次函数图像平移——只动顶点，不动 a}


\begin{eduSolutionBox}{核心结论：顶点式平移——只动顶点，不动 a}
  \begingroup
  \setlength{\parskip}{2.5mm}
顶点式 $y=a(x-h)^2+k$，顶点为 $(h,k)$，开口方向与宽窄由 $a$ 决定。\par
\textbf{平移只改变顶点位置，不改变 $a$。}\par
\begin{tabular}{p{0.30\linewidth}p{0.58\linewidth}}
平移方向 & 代换动作\\
向左 $m$ 个单位 & 括号内 $x$ 换成 $x+m$\\
向右 $m$ 个单位 & 括号内 $x$ 换成 $x-m$\\
向上 $n$ 个单位 & $k$ 换成 $k+n$\\
向下 $n$ 个单位 & $k$ 换成 $k-n$
\end{tabular}
\par
口诀：\textbf{左加右减（对 $x$），上加下减（对 $k$）}；$a$ 原样抄。\par
  \endgroup
\end{eduSolutionBox}




\begin{eduProblemBlock}[例题 1（A 类：已知原解析式 + 平移 → 求新解析式）]
将抛物线 $y=2(x-3)^2+1$ 先向左平移 $2$ 个单位，再向上平移 $3$ 个单位，求平移后得到的新解析式。

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
读原顶点与 $a$$\;\to\;$向左 $2$：$x\to x+2$$\;\to\;$向上 $3$：$k\to k+3$\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 1}
求 $y=2(x-3)^2+1$ 向左 $2$、向上 $3$ 后的新解析式。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideMistake{左右符号}{向左是“加”：$x\to x+2$。如果你写成了 $x-2$，方向就反了。}
        \eduSideHint{为什么 $a$ 不动}{平移只移动整条抛物线的位置，不改变开口方向和宽窄，所以 $a=2$ 原样抄。}
        \eduSideHint{怎么验算}{原顶点 $(3,1)$ 左 $2$、上 $3$ 应得到 $(1,4)$，与新解析式顶点对得上吗？}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{读原顶点与 $a$}{原式 $y=2(x-3)^2+1$，顶点为 $(3,1)$，$a=2$。}
      \eduExplainStep{2}{向左 $2$：$x\to x+2$}{向左平移 $2$ 个单位，把括号内的 $x$ 换成 $x+2$：$2(x-3+2)^2+1=2(x-1)^2+1$。}
      \eduExplainStep{3}{向上 $3$：$k\to k+3$}{向上平移 $3$ 个单位，把 $k$ 换成 $k+3$：$y=2(x-1)^2+(1+3)=2(x-1)^2+4$，新顶点为 $(1,4)$，$a=2$ 不变。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduProblemBlock}[例题 2（B 类：已知原解析式 + 新解析式 → 求平移方式）]
已知抛物线 $y=-(x+1)^2+2$ 经过平移后得到 $y=-(x-3)^2+5$。说明这条抛物线是怎样平移的（向哪、几个单位）。

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
分别读出原、新顶点$\;\to\;$求 $\Delta x$ 并判左右$\;\to\;$求 $\Delta y$ 并判上下\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 2}
由 $y=-(x+1)^2+2$ 到 $y=-(x-3)^2+5$ 是怎样平移的？\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{先想什么}{$a$ 没变说明只是平移；只要盯住顶点，比较新旧顶点坐标差即可。}
        \eduSideMistake{方向判反}{$\Delta x>0$ 是向右，$\Delta x<0$ 是向左；别看到 $x+1$ 里有“$+$”就以为是向左。}
        \eduSideHint{怎么验算}{把 $(-1,2)$ 右 $4$、上 $3$ 后，是否落到 $(3,5)$？}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{分别读出原、新顶点}{原式顶点 $(-1,2)$，新式顶点 $(3,5)$；两式 $a=-1$ 相同。}
      \eduExplainStep{2}{求 $\Delta x$ 并判左右}{$\Delta x=3-(-1)=4>0$，顶点横坐标增大 $4$，即向右平移 $4$ 个单位。}
      \eduExplainStep{3}{求 $\Delta y$ 并判上下}{$\Delta y=5-2=3>0$，顶点纵坐标增大 $3$，即向上平移 $3$ 个单位。}
    \vspace{1mm}
    {\color{edu-blue}\bfseries\small 结论}\par\vspace{1mm}
    \begin{eduBulletList}
      \item 向右平移 $4$ 个单位，再向上平移 $3$ 个单位。    \end{eduBulletList}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduProblemBlock}[例题 3（C 类：已知新解析式 + 平移方式 → 反推原解析式）]
一条抛物线经过“向左平移 $3$ 个单位、向下平移 $2$ 个单位”后，得到新解析式 $y=3(x-2)^2-1$。求原解析式。

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
读新顶点与 $a$$\;\to\;$反推原顶点横坐标 $h$$\;\to\;$反推原顶点纵坐标 $k$$\;\to\;$写出原顶点式\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 3}
已知新解析式 $y=3(x-2)^2-1$ 由原式向左 $3$、向下 $2$ 得到，求原解析式。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{先想什么}{“原经平移得新”就是：新顶点 $=$ 原顶点 $+$ 平移量。把它反过来解原顶点。}
        \eduSideMistake{方向搞反}{向左 $3$ 意味着新横坐标 $=h-3$（不是 $h+3$）；向下 $2$ 意味着新纵坐标 $=k-2$。}
        \eduSideHint{怎么验算}{把求出的原式再向左 $3$、向下 $2$，能否回到 $y=3(x-2)^2-1$？}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{读新顶点与 $a$}{新式 $y=3(x-2)^2-1$，新顶点 $(2,-1)$，$a=3$。}
      \eduExplainStep{2}{反推原顶点横坐标 $h$}{原经“左 $3$”到新，故新横坐标 $=h-3=2$，解得 $h=5$。}
      \eduExplainStep{3}{反推原顶点纵坐标 $k$}{原经“下 $2$”到新，故新纵坐标 $=k-2=-1$，解得 $k=1$。}
      \eduExplainStep{4}{写出原顶点式}{原顶点 $(5,1)$，$a=3$ 不变，原式 $y=3(x-5)^2+1$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduMistakeBox}
\eduSubTitle{易错提醒}\begin{itemize}
  \item \textbf{左右符号反}：向左 $m$ 是 $x\to x+m$（加），向右 $m$ 才是 $x\to x-m$。陆子辰最易把“向左”写成 $x-m$。
  \item \textbf{动了 $a$}：平移不改变 $a$，不要把 $a$ 也加减。
  \item \textbf{左右量加到 $k$}：左右只改括号内的 $x$，上下才改 $k$。
  \item \textbf{先展开再平移}：左右平移只对括号内的 $x$ 操作，不要先展开成一般式再乱加。
\end{itemize}
\end{eduMistakeBox}




\begin{eduSummaryBlock}{\eduIconPin}{方法提醒}
\begin{eduBulletList}
  \item 三要素：顶点 $(h,k)$、$a$、平移方向与单位数；知二推一。  \item A 类顺推：对 $x$、$k$ 做平移代换；B 类对比：比新旧顶点差；C 类逆推：由新顶点反解原顶点。  \item 口诀：左加右减（对 $x$），上加下减（对 $k$），$a$ 永远不动。\end{eduBulletList}
\end{eduSummaryBlock}
