\eduSectionTitle{知识点 3：一般式求对称轴与顶点（公式法）}


\begin{eduSolutionBox}{核心结论：一般式求对称轴与顶点的公式}
  \begingroup
  \setlength{\parskip}{2.5mm}
一般式 $y=ax^2+bx+c$（$a\ne0$）的对称轴为 $x=h=-\dfrac{b}{2a}$（\textbf{注意负号}）。\par
顶点纵坐标 $k$：把 $x=h$ 代入原式计算，或用公式 $k=\dfrac{4ac-b^2}{4a}$。\par
顶点为 $\left(-\dfrac{b}{2a},\,\dfrac{4ac-b^2}{4a}\right)$。\par
\begin{tabular}{p{0.18\linewidth}p{0.30\linewidth}p{0.40\linewidth}}
步骤 & 动作 & 关键点\\
1. 求对称轴 & $h=-\dfrac{b}{2a}$ & 负号不可漏\\
2. 求顶点纵坐标 & 把 $x=h$ 代入原式 & 比 $k$ 公式更不易错\\
3. 写结论 & 对称轴 $x=h$，顶点 $(h,k)$ & 两者共用 $h$\\
\end{tabular}
\par
  \endgroup
\end{eduSolutionBox}




\begin{eduProblemBlock}[例题 1（a=1）]
求二次函数 $y=x^2-4x+3$ 的对称轴和顶点坐标。

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
用 $h=-\dfrac{b}{2a}$ 求对称轴$\;\to\;$把 $x=h$ 代入原式求 $k$$\;\to\;$写出顶点\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 1}
求 $y=x^2-4x+3$ 的对称轴和顶点。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideMistake{负号别漏}{对称轴是 $-\dfrac{-4}{2}=2$，分子上的 $-4$ 和公式前面的负号抵消得 $2$；不要写成 $-2$。}
        \eduSideHint{为什么代入而不是背 $k$ 公式}{代入只做一次加乘，比 $k=\dfrac{4ac-b^2}{4a}$ 符号更少，更不易错。}
        \eduSideHint{公式反验}{$k=\dfrac{4\cdot1\cdot3-(-4)^2}{4}=\dfrac{12-16}{4}=-1$，与代入结果一致。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{用 $h=-\dfrac{b}{2a}$ 求对称轴}{$h=-\dfrac{-4}{2\cdot1}=2$，对称轴 $x=2$。}
      \eduExplainStep{2}{把 $x=h$ 代入原式求 $k$}{$k=2^2-4\cdot2+3=4-8+3=-1$。}
      \eduExplainStep{3}{写出顶点}{顶点为 $(2,-1)$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduProblemBlock}[例题 2（a≠1）]
求二次函数 $y=2x^2+8x+5$ 的对称轴和顶点坐标。

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
用 $h=-\dfrac{b}{2a}$ 求对称轴$\;\to\;$把 $x=h$ 代入原式求 $k$$\;\to\;$写出顶点\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 2}
求 $y=2x^2+8x+5$ 的对称轴和顶点。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideMistake{分母别忘乘 a}{分母是 $2a=2\cdot2=4$，不是 $2$；$-\dfrac{8}{4}=-2$。}
        \eduSideMistake{代入时 $(-2)^2=4$}{代入 $x=-2$ 时，平方项 $2\cdot(-2)^2=2\cdot4=8$，不是 $-8$。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{用 $h=-\dfrac{b}{2a}$ 求对称轴}{$h=-\dfrac{8}{2\cdot2}=-2$，对称轴 $x=-2$。}
      \eduExplainStep{2}{把 $x=h$ 代入原式求 $k$}{$k=2\cdot(-2)^2+8\cdot(-2)+5=8-16+5=-3$。}
      \eduExplainStep{3}{写出顶点}{顶点为 $(-2,-3)$。}
    \vspace{1mm}
    {\color{edu-blue}\bfseries\small 验算}\par\vspace{1mm}
    \begin{eduBulletList}
      \item $k=\dfrac{4\cdot2\cdot5-8^2}{4\cdot2}=\dfrac{40-64}{8}=-3$，与代入结果一致。    \end{eduBulletList}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduProblemBlock}[例题 3（a 为负）]

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}

\begin{eduSubQuestionItem}{例题 3}
求 $y=-x^2+6x-7$ 的对称轴和顶点。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideMistake{$a$ 为负时分母为负}{$a=-1$，所以 $2a=-2$，$-\dfrac{6}{-2}=3$；负负得正，结果为正。}
        \eduSideMistake{代入时负号别漏}{$-x^2$ 代入 $3$ 是 $-3^2=-9$，不是 $9$；符号在最前面。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{用 $h=-\dfrac{b}{2a}$ 求对称轴}{$h=-\dfrac{6}{2\cdot(-1)}=-\dfrac{6}{-2}=3$，对称轴 $x=3$。}
      \eduExplainStep{2}{把 $x=h$ 代入原式求 $k$}{$k=-3^2+6\cdot3-7=-9+18-7=2$。}
      \eduExplainStep{3}{写出顶点}{顶点为 $(3,2)$。}
    \vspace{1mm}
    {\color{edu-blue}\bfseries\small 验算}\par\vspace{1mm}
    \begin{eduBulletList}
      \item $k=\dfrac{4\cdot(-1)\cdot(-7)-6^2}{4\cdot(-1)}=\dfrac{28-36}{-4}=2$，与代入结果一致。    \end{eduBulletList}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduMistakeBox}
\eduSubTitle{易错提醒}\begin{itemize}
  \item \textbf{漏负号}：对称轴公式 $-\dfrac{b}{2a}$，负号在分子前，不可省略。
  \item \textbf{分母漏乘 $a$}：分母是 $2a$ 不是 $2$。
  \item \textbf{$a$ 为负时分母为负}：$-\dfrac{b}{2a}$ 中 $a$ 的符号要带进分母，负负得正。
  \item \textbf{代入算 $k$ 时漏首项负号}：$-x^2$ 代入是负的，不是正的。
\end{itemize}
\end{eduMistakeBox}




\begin{eduSummaryBlock}{\eduIconPin}{方法提醒}
\begin{eduBulletList}
  \item 对称轴 $x=-\dfrac{b}{2a}$（负号不可漏）。  \item 顶点纵坐标 $k$：代入 $x=h$ 到原式算，比 $k$ 公式更稳。  \item 顶点 $(h,k)$，对称轴 $x=h$，共用 $h$。  \item 用 $k=\dfrac{4ac-b^2}{4a}$ 反验。\end{eduBulletList}
\end{eduSummaryBlock}
