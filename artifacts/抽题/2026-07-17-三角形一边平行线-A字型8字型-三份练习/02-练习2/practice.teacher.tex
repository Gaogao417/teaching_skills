\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    如图，$P,A,C$ 三点共线，$P,B,D$ 三点共线，点序分别为 $P-A-C$、$P-B-D$，且 $AB\parallel CD$。
已知 $PA=\frac{6}{7}$，$PC=\frac{36}{7}$，$CD=\frac{72}{7}$，求 $AB$。
    
  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{先看平行边方向，再判断 A 字型或 8 字型及对应边。}
  \input{\detokenize{../../../题库/2026-07-17-三边求第四边-A字型8字型/items/Q031/build/diagram/jobs/question-bank-three-known-q031-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[42mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$AB=\frac{12}{7}$。\par\medskip
    \solstep{先判断}{已知边与未知边中出现平行线段 $CD$、$AB$，先求两个相似三角形的比。}
    \solstep{计算并约分}{$\dfrac{PA}{PC}=\dfrac{\frac{6}{7}}{\frac{36}{7}}=\dfrac{1}{6}$。}
    \solstep{标份数}{约分后，$AB$ 对应 $1$ 份，$CD$ 对应 $6$ 份。}
    \solstep{按份数公式求边}{$AB=CD\times\dfrac{1}{6}=\frac{72}{7}\times\dfrac{1}{6}=\frac{12}{7}$。}
  \end{solutionblock}
\end{problem}
