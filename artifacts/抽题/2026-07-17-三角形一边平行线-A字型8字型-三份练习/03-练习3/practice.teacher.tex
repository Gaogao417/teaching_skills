\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    如图，$P,A,C$ 三点共线，$P,B,D$ 三点共线，点序分别为 $P-A-C$、$P-B-D$，且 $AB\parallel CD$。
已知 $PB=\frac{1}{8}$，$PD=\frac{5}{8}$，$PC=\frac{15}{8}$，求 $PA$。
    
  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{先看平行边方向，再判断 A 字型或 8 字型及对应边。}
  \input{\detokenize{../../../题库/2026-07-17-三边求第四边-A字型8字型/items/Q041/build/diagram/jobs/question-bank-three-known-q041-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[42mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$PA=\frac{3}{8}$。\par\medskip
    \solstep{先判断}{已知边与未知边都不在平行线段 $AB,CD$ 上，先求两侧对应线段的比。}
    \solstep{计算并约分}{$\dfrac{PB}{PD}=\dfrac{\frac{1}{8}}{\frac{5}{8}}=\dfrac{1}{5}$。}
    \solstep{标份数}{约分后，$PA$ 对应 $1$ 份，$PC$ 对应 $5$ 份。}
    \solstep{按份数公式求边}{$PA=PC\times\dfrac{1}{5}=\frac{15}{8}\times\dfrac{1}{5}=\frac{3}{8}$。}
  \end{solutionblock}
\end{problem}
