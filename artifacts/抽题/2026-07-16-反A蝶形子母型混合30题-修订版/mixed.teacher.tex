\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    如图，$\angle PAB=\angle PDC$。

已知 $AB=\frac{10}{9}$，$DC=\frac{50}{9}$，$PA=\frac{10}{9}$，求 $PD$。
    
  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{沿题设等角的顶点顺序，找出两组三角形的对应边。}
  \input{\detokenize{../../题库/2026-07-16-反A形相似/items/Q017/build/diagram/jobs/question_bank-reverse_a-q017-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[48mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$PD=\frac{50}{9}$。\par\medskip
    \solstep{见等角，找相似}{由题设等角和构型自带的第二组等角，得 $\triangle PAB\sim\triangle PDC$。}
    \solstep{按顶点写对应边}{三组小边与大边依次为：$PA\leftrightarrow PD$，$PB\leftrightarrow PC$，$AB\leftrightarrow DC$。}
    \solstep{保持比例方向一致}{取 小三角形边与大三角形对应边之比，可列 $\dfrac{AB}{DC}=\dfrac{PA}{PD}$。}
    \solstep{代入并验算}{代入已知量解得 $PD=\frac{50}{9}$；两组对应边的比均等于同一相似比。}
  \end{solutionblock}
\end{problem}
