\begin{question}[points=4]
  已知点 $P(2a-1,a+1)$，如果点 $Q$ 的坐标为 $(5,2)$，且直线 $PQ\parallel y$ 轴，那么点 $P$ 的坐标是\underline{\hspace{22mm}}。
  \paren[$(5,4)$]
\end{question}

\begin{problem}[points=4]
  如图，在平面直角坐标系中，矩形 $ABCD$ 的对角线 $AC,BD$ 相交于点 $E$，$B(-1,0)$，$C(0,\sqrt3)$，如果 $AC\parallel x$ 轴，那么 $BE$ 的长为\underline{\hspace{18mm}}。
  \begin{answerareawithdiagramtikz}{68mm}{60mm}{矩形与坐标轴}
  \input{\detokenize{build/diagram/jobs/q14-prompt/rendered/prompt.fragment.tex}}
  \end{answerareawithdiagramtikz}

  \begin{solutionblock}
    \textbf{答案：}$2$\par\medskip
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=4]
  如图，$\triangle ABC$ 中，$\angle BAC=90^\circ$，$\angle ABC=30^\circ$。将 $\triangle ABC$ 绕点 $A$ 按顺时针方向旋转到 $\triangle ADE$（点 $D$ 与点 $B$ 对应），且使射线 $AD\parallel BC$，直线 $DE$ 交射线 $BC$ 于点 $G$，那么 $\angle EAG$ 的度数为\underline{\hspace{18mm}}度。

  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=4]
  如图，在 $\triangle ABC$ 中，点 $E,F$ 分别是 $AB,BC$ 的中点，$EH\perp AC$ 于 $H$，$GF\perp BC$ 且交 $EH$ 于点 $G$，$GF=FC$。若 $BG=\sqrt5$，$AH=3$，则 $CH$ 的长度为\underline{\hspace{18mm}}。
  \begin{answerareawithdiagramtikz}{64mm}{60mm}{中点与垂线}

  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=8]
  如图，在平面直角坐标系中，四边形 $OABC$ 是平行四边形。点 $O$ 是原点，点 $A$ 的坐标是 $(-16,0)$。线段 $BC$ 交 $y$ 轴于点 $D(0,8)$，$CD=6$。动点 $P$ 从点 $O$ 出发，沿射线 $OA$ 的方向以每秒 $2$ 个单位的速度运动，同时动点 $Q$ 从点 $D$ 出发，以每秒 $1$ 个单位的速度向点 $B$ 运动，当点 $Q$ 运动到点 $B$ 时，点 $P$ 随之停止运动。设运动时间为 $t$ 秒。\par (1) 用含 $t$ 的代数式表示：$BQ=$\underline{\hspace{18mm}}；\par (2) 当以 $P,Q,A,B$ 为顶点的四边形是平行四边形时，则 $t$ 的值是\underline{\hspace{18mm}}。
  \begin{answerareawithdiagramtikz}{44mm}{60mm}{坐标系中的平行四边形}
  \input{\detokenize{build/diagram/jobs/q19-prompt/rendered/prompt.fragment.tex}}
  \end{answerareawithdiagramtikz}
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}(1) $10-t$；(2) $\dfrac{26}{3}$\par\medskip
  \end{solutionblock}
\end{problem}
