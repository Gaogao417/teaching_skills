\begin{eduSolutionBox}{三条主线：从属关系（基本事实2是桥）／公共点→交线（基本事实3）／定面（基本事实1+推论①②③）}
  \begingroup
  \setlength{\parskip}{2.5mm}
\textbf{主线一·从属关系：}点属于线、线属于面，靠基本事实2搭桥。$A\in l$，$B\in l$，$A\in\alpha$，$B\in\alpha\Rightarrow l\subset\alpha$。即“点在线上 + 两点在面内 $\Rightarrow$ 整条线在面内”。\par
\textbf{主线二·公共点$\to$交线：}两个不重合的平面有一个公共点 $\Rightarrow$ 有且只有一条过该点的交线。$P\in\alpha$，$P\in\beta\Rightarrow\alpha\cap\beta=l$，$P\in l$。这是“面和面怎么发生关系”的入口。\par
\textbf{主线三·定面：}基本事实1（3 个不共线点$\Rightarrow$平面）+ 推论①（直线+外一点）、推论②（两条相交直线）、推论③（两条平行直线），四条都在回答“多少东西能确定一个平面”。\par
\begin{tabular}{p{0.20\linewidth}p{0.62\linewidth}}
\textbf{符号} & \textbf{含义}\\
$A\in l$ & 点 $A$ 在直线 $l$ 上\\
$l\subset\alpha$ & 直线 $l$ 在平面 $\alpha$ 内（$\alpha$ 经过 $l$）\\
$\alpha\cap\beta=l$ & 平面 $\alpha$ 与 $\beta$ 相交于直线 $l$\\
\end{tabular}
\par
\begin{tabular}{p{0.18\linewidth}p{0.24\linewidth}p{0.40\linewidth}}
\textbf{分类} & \textbf{对象} & \textbf{三类情况}\\
线线 & 直线与直线 & 相交 / 平行 / 异面\\
线面 & 直线与平面 & 在面内 / 相交 / 平行\\
面面 & 平面与平面 & 平行 / 相交\\
\end{tabular}
\par
  \endgroup
\end{eduSolutionBox}




\begin{eduMistakeBox}
\eduSubTitle{易错提醒}别把“从属关系（$\in/\subset$，讲归属）”和“位置关系分类（相交/平行/异面，讲怎么摆）”混在一起。判断时先问一句：这条定理讲的是“谁属于谁”还是“怎么摆”？基本事实 2 是从属关系的桥，基本事实 3 是交线的入口。\end{eduMistakeBox}

\eduSectionTitle{B 类定理卡·平行：给图、一步推理、补中间条件}


\begin{eduSolutionBox}{定理卡·线面平行判定：在面内找平行线}
  \begingroup
  \setlength{\parskip}{2.5mm}
\textbf{几何语言：}$a\not\subset\alpha$，$b\subset\alpha$，$a\parallel b\Rightarrow a\parallel\alpha$。\par
\textbf{口令：}线线平行 $\Rightarrow$ 线面平行。在面内找一条与已知直线平行的线。\par
\textbf{易错前提：}$a\not\subset\alpha$（线在面外）与 $b\subset\alpha$（线在面内）两条缺一不可；少写“平面内”判定不成立。\par
  \endgroup
\end{eduSolutionBox}

\begin{eduSubQuestionItem}{例1·线面平行判定}
已知 $a\parallel b$，$b\subset\alpha$，补出中间前提后证明 $a\parallel\alpha$。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{先判对象}{这是由线线平行推线面平行，属于线面平行判定。}
        \eduSideMistake{最易漏}{最容易漏 $a\not\subset\alpha$，没写“线在面外”判定不成立。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{判对象}{线线（$a\parallel b$）$\to$ 线面（$a\parallel\alpha$），用线面平行判定。}
      \eduExplainStep{2}{补中间前提}{补 $a\not\subset\alpha$（线在面外），与 $b\subset\alpha$、$a\parallel b$ 合起来。}
      \eduExplainStep{3}{写结论}{$a\not\subset\alpha$，$b\subset\alpha$，$a\parallel b\Rightarrow a\parallel\alpha$。}
  \end{eduExplainRight}
\end{eduDualExplain}





\begin{eduSolutionBox}{定理卡·线面平行性质：过线作面找交线}
  \begingroup
  \setlength{\parskip}{2.5mm}
\textbf{几何语言：}$a\parallel\alpha$，$a\subset\beta$，$\alpha\cap\beta=b\Rightarrow a\parallel b$。\par
\textbf{口令：}线面平行 $\Rightarrow$ 线线平行。过该直线作辅助面，找它与已知面的交线。\par
\textbf{易错前提：}$a\subset\beta$ 与 $\alpha\cap\beta=b$（交线）是最常漏的“桥”，不能直接由 $a\parallel\alpha$ 跳到 $a\parallel b$。\par
  \endgroup
\end{eduSolutionBox}

\begin{eduSubQuestionItem}{例2·线面平行性质}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{这步找什么}{线面平行要推出线线平行，必须靠“辅助面+交线”搭桥。}
        \eduSideMistake{别跳步}{不能直接由 $AB\parallel\alpha$ 写 $AB\parallel CD$，必须补 $\alpha\cap\beta=CD$。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{判对象}{线面平行（$AB\parallel\alpha$）$\to$ 线线平行（$AB\parallel CD$），用线面平行性质。}
      \eduExplainStep{2}{补交线（桥）}{补 $AB\subset\beta$，$\alpha\cap\beta=CD$，这一步是桥，不能跳。}
      \eduExplainStep{3}{写结论}{$AB\parallel\alpha$，$AB\subset\beta$，$\alpha\cap\beta=CD\Rightarrow AB\parallel CD$。}
  \end{eduExplainRight}
\end{eduDualExplain}





\begin{eduSolutionBox}{定理卡·面面平行判定：一面内两条相交线都平行另一面}
  \begingroup
  \setlength{\parskip}{2.5mm}
\textbf{几何语言：}$a\subset\alpha$，$b\subset\alpha$，$a\cap b=A$，$a\parallel\beta$，$b\parallel\beta\Rightarrow\alpha\parallel\beta$。\par
\textbf{口令：}线面平行 $\Rightarrow$ 面面平行。在一个面内找两条相交线，分别平行另一面。\par
\textbf{易错前提：}“两条相交线”$a\cap b=A$ 是头号失分点，写两条平行线不行。\par
  \endgroup
\end{eduSolutionBox}

\begin{eduSubQuestionItem}{例3·面面平行判定}
已知 $\alpha$ 内直线 $a,b$ 都平行 $\beta$，补相交前提后证明 $\alpha\parallel\beta$。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{一面内几条线}{面面平行判定要在一个面内找两条线，不是一条。}
        \eduSideMistake{相交不能漏}{$a\cap b=A$ 是头号失分点，两条平行线推不出面面平行。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{判对象}{线面平行 $\to$ 面面平行，用面面平行判定。}
      \eduExplainStep{2}{补相交前提}{补 $a\cap b=A$，即这两条面内直线必须相交。}
      \eduExplainStep{3}{写结论}{$a\subset\alpha$，$b\subset\alpha$，$a\cap b=A$，$a\parallel\beta$，$b\parallel\beta\Rightarrow\alpha\parallel\beta$。}
  \end{eduExplainRight}
\end{eduDualExplain}





\begin{eduSolutionBox}{定理卡·面面平行性质：第三个面截出两条平行交线}
  \begingroup
  \setlength{\parskip}{2.5mm}
\textbf{几何语言：}$\alpha\parallel\beta$，$\alpha\cap\gamma=a$，$\beta\cap\gamma=b\Rightarrow a\parallel b$。\par
\textbf{口令：}面面平行 $\Rightarrow$ 线线平行。两平行面被第三个面截，两条交线平行。\par
\textbf{易错前提：}要写清第三个面 $\gamma$，以及 $\alpha\cap\gamma=a$、$\beta\cap\gamma=b$ 两条交线。\par
  \endgroup
\end{eduSolutionBox}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{关键动作}{面面平行推线线平行，靠的是“第三个面截出两条交线”。}
        \eduSideMistake{交线要写全}{两条交线 $\alpha\cap\gamma=a$、$\beta\cap\gamma=b$ 都要写，缺一条结论不成立。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{判对象}{面面平行（$\alpha\parallel\beta$）$\to$ 线线平行（$a\parallel b$），用面面平行性质。}

      \eduExplainStep{3}{写结论}{$\alpha\parallel\beta$，$\alpha\cap\gamma=a$，$\beta\cap\gamma=b\Rightarrow a\parallel b$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduMistakeBox}
\eduSubTitle{平行定理易错提醒}平行类最常漏两类前提：判定定理漏“相交”（$a\cap b=A$）或“平面内/面外”（$a\not\subset\alpha$、$b\subset\alpha$）；性质定理漏“辅助面+交线”。补中间条件时先问：这座桥是交线，还是面内/相交前提？\end{eduMistakeBox}

\eduSectionTitle{B 类定理卡·垂直：给图、一步推理、补中间条件}


\begin{eduSolutionBox}{定理卡·线面垂直判定：面内两条相交线都垂直}
  \begingroup
  \setlength{\parskip}{2.5mm}
\textbf{几何语言：}$l\perp m$，$l\perp n$，$m\subset\alpha$，$n\subset\alpha$，$m\cap n=A\Rightarrow l\perp\alpha$。\par
\textbf{口令：}线线垂直 $\Rightarrow$ 线面垂直。在面内找两条与已知直线都垂直的相交线。\par
\textbf{易错前提：}$m\cap n=A$（两条相交线）是头号失分点；两条不相交的线垂直推不出线面垂直。\par
  \endgroup
\end{eduSolutionBox}

        \eduSideMistake{头号失分点}{$m\cap n=A$ 最易漏，两条不相交的线推不出线面垂直。}

\begin{eduSolutionBox}{定理卡·线面垂直性质：同垂直一面则两线平行}
  \begingroup
  \setlength{\parskip}{2.5mm}
\textbf{几何语言：}$a\perp\alpha$，$b\perp\alpha\Rightarrow a\parallel b$。\par
\textbf{口令：}线面垂直 $\Rightarrow$ 线线平行。两条都垂直同一个面的线互相平行（跨类结论）。\par
\textbf{易错前提：}必须同一个平面 $\alpha$；若垂直不同的面，结论不成立。\par
  \endgroup
\end{eduSolutionBox}

\begin{eduSolutionBox}{定理卡·面面垂直判定：一面过另一面的垂线}
  \begingroup
  \setlength{\parskip}{2.5mm}
\textbf{几何语言：}$l\perp\alpha$，$l\subset\beta\Rightarrow\alpha\perp\beta$。\par
\textbf{口令：}线面垂直 $\Rightarrow$ 面面垂直。一个面内有一条线垂直另一个面。\par
\textbf{易错前提：}垂线 $l$ 必须在某一面内（$l\subset\beta$），且垂直另一面（$l\perp\alpha$），两条缺一不可。\par
  \endgroup
\end{eduSolutionBox}

\begin{eduSubQuestionItem}{例7·面面垂直判定}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{找什么}{面面垂直判定要在一个面内找一条垂直另一个面的线。}
        \eduSideMistake{两条前提}{$l\perp\alpha$ 和 $l\subset\beta$ 都要写，少一条结论不成立。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{判对象}{线面垂直（$l\perp\alpha$）$\to$ 面面垂直（$\alpha\perp\beta$），用面面垂直判定。}
      \eduExplainStep{2}{补垂线在面内}{补 $l\subset\beta$，垂线必须在某一个面内。}
      \eduExplainStep{3}{写结论}{$l\perp\alpha$，$l\subset\beta\Rightarrow\alpha\perp\beta$。}
  \end{eduExplainRight}
\end{eduDualExplain}





\begin{eduSolutionBox}{定理卡·面面垂直性质：四件套（面面垂直+交线+面内+垂直于棱）}
  \begingroup
  \setlength{\parskip}{2.5mm}
\textbf{几何语言：}$\alpha\perp\beta$，$\alpha\cap\beta=l$，$AB\subset\alpha$，$AB\perp l\Rightarrow AB\perp\beta$。\par
\textbf{口令：}面面垂直 $\Rightarrow$ 线面垂直。一个面内垂直于交线（棱）的线，垂直另一个面。\par
\textbf{易错前提（四件套缺一不可）：}①$\alpha\perp\beta$；②$\alpha\cap\beta=l$（交线/棱）；③$AB\subset\alpha$（线在面内）；④$AB\perp l$（垂直于棱）。\par
  \endgroup
\end{eduSolutionBox}

\begin{eduSolutionBox}{定理卡·二面角平面角定义：顶点在棱上、两边在两面内、两边垂直于棱}
  \begingroup
  \setlength{\parskip}{2.5mm}
\textbf{几何语言：}$\alpha\cap\beta=l$，$O\in l$，$OA\subset\alpha$，$OB\subset\beta$，$OA\perp l$，$OB\perp l\Rightarrow\angle AOB$ 为二面角 $\alpha\!-\!l\!-\!\beta$ 的平面角。\par
\textbf{口令：}三要素——顶点在棱上、两边分别在两个半平面内、两边都垂直于棱。\par
\textbf{易错点：}平面角的顶点 $O$ 必须在棱 $l$ 上；$OA,OB$ 必须都垂直于棱 $l$，且分别在两个面内。\par
  \endgroup
\end{eduSolutionBox}
