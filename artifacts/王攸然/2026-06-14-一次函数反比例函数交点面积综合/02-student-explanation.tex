\begin{eduProblemBlock}[题目]
\noindent
\begin{minipage}[t]{\dimexpr\linewidth-65mm-6mm\relax}
\vspace{0pt}
已知一次函数 $y_1=kx+b$ 与反比例函数 $y_2=\dfrac{m}{x}$ 的图像交于点 $A,B$。已知：点 $A$ 的横坐标为 $2$，点 $B$ 的横坐标为 $-1$，一次函数图像与 $y$ 轴交于点 $C(0,-1)$。
\begin{enumerate}[label=(\arabic*)]
  \item 求一次函数解析式、反比例函数解析式，以及点 $A,B$ 的坐标；
  \item 根据图像求不等式 $kx+b>\dfrac{m}{x}$ 的解集；
  \item 求 $\triangle OAB$ 的面积。
\end{enumerate}

\end{minipage}\hfill
\diagramcolfigure{diagrams/original.pdf}{65mm}{原题函数图像}
\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
先由截距确定 $b$$\;\to\;$用 $k$ 表示交点坐标$\;\to\;$用 $xy=m$ 列方程$\;\to\;$求出函数和交点$\;\to\;$图像分段解不等式$\;\to\;$用铅垂线法求面积\end{eduAuxItem}




\begin{eduSubQuestionItem}{(1)}
求一次函数解析式、反比例函数解析式，以及点 $A,B$ 的坐标。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{提示与易错}
        \eduSideHint{第一步}{$C(0,-1)$ 直接给出直线的截距，不要先列复杂方程组。}
        \eduSideHint{交点关系}{点在 $y=\dfrac{m}{x}$ 上，就有 $xy=m$。两个交点的 $xy$ 应相等。}
        \eduSideMistake{不要漏写坐标}{求出函数后，要把 $A,B$ 的完整坐标写出来，不只写横坐标。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{先由截距确定 $b$}{由 $C(0,-1)$ 在直线 $y=kx+b$ 上，得 $b=-1$。}
      \eduExplainStep{2}{用 $k$ 表示交点坐标}{直线为 $y=kx-1$，所以 $A(2,2k-1)$，$B(-1,-k-1)$。}
      \eduExplainStep{3}{用 $xy=m$ 列方程}{$A,B$ 都在 $y=\dfrac{m}{x}$ 上，所以 $2(2k-1)=(-1)(-k-1)$。}
      \eduExplainStep{4}{求出函数和交点}{解得 $k=1$，所以 $y_1=x-1$，$A(2,1)$，$B(-1,-2)$，$m=2$，$y_2=\dfrac2x$。}
    \vspace{1mm}
    {\color{edu-blue}\bfseries\small 验算}\par\vspace{1mm}
    \begin{eduBulletList}
      \item $A(2,1)$ 满足 $1=2-1$，且 $1=\dfrac22$。      \item $B(-1,-2)$ 满足 $-2=-1-1$，且 $-2=\dfrac2{-1}$。    \end{eduBulletList}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduSubQuestionItem}{(2)}
根据图像求不等式 $kx+b>\dfrac{m}{x}$ 的解集。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{提示与易错}
        \eduSideHint{先翻译}{$kx+b>\dfrac{m}{x}$ 就是 $y_1>y_2$，图像上看直线是否在双曲线上方。}
        \eduSideMistake{$0$ 也要分段}{反比例函数在 $x=0$ 无定义，所以 $0$ 必须作为断点，且不能取。}
        \eduSideMistake{严格大于}{交点处两函数相等，题目是 $>$，所以 $x=-1,2$ 都不能取。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{图像分段解不等式}{$x-1>\dfrac2x$ 表示直线在双曲线上方；用 $-1,0,2$ 分段，取 $(-1,0)$ 和 $(2,+\infty)$。}
    \vspace{1mm}
    {\color{edu-blue}\bfseries\small 图像校对}\par\vspace{1mm}
    \begin{eduBulletList}
      \item $(-\infty,-1)$：双曲线在上，不取。      \item $(-1,0)$：直线在上，取。      \item $(0,2)$：双曲线在上，不取。      \item $(2,+\infty)$：直线在上，取。      \item 解集为 $-1<x<0$ 或 $x>2$。    \end{eduBulletList}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduSubQuestionItem}{(3)}
求 $\triangle OAB$ 的面积。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{提示与易错}
        \eduSideHint{铅垂线法}{先看 $AB$ 与 $y$ 轴的交点 $P$，把 $\triangle OAB$ 拆成两个共用高 $OP$ 的小三角形。}
        \eduSideHint{公式怎么来}{$S_{\triangle OAB}=S_{\triangle OPA}+S_{\triangle OPB}=\dfrac12 OP\cdot|x_A|+\dfrac12 OP\cdot|x_B|$；本题 $A,B$ 在 $y$ 轴两侧，所以 $|x_A|+|x_B|=|x_A-x_B|$。}
        \eduSideMistake{别漏掉 $\dfrac12$}{铅垂线法的面积仍是三角形面积，公式是 $\dfrac12\cdot OP\cdot|x_A-x_B|$。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{用铅垂线法求面积}{直线 $AB$ 与 $y$ 轴交于 $P(0,-1)$，所以 $OP=1$。把 $\triangle OAB$ 按 $OP$ 拆成 $\triangle OPA$ 和 $\triangle OPB$，两者共用底 $OP$，高分别是 $|x_A|$、$|x_B|$。因为 $A,B$ 在 $y$ 轴两侧，所以 $S_{\triangle OAB}=S_{\triangle OPA}+S_{\triangle OPB}=\dfrac12OP(|x_A|+|x_B|)=\dfrac12OP|x_A-x_B|=\dfrac12\cdot1\cdot|2-(-1)|=\dfrac32$。}
    \vspace{1mm}
    {\color{edu-blue}\bfseries\small 结论}\par\vspace{1mm}
    \begin{eduBulletList}
      \item $S_{\triangle OAB}=\dfrac32$。    \end{eduBulletList}
  \end{eduExplainRight}
\end{eduDualExplain}
