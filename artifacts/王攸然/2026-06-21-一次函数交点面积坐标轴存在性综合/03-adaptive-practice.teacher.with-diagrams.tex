\section*{一次函数交点、面积与坐标轴存在性综合 · 教师版}

\section*{练习题（每题 10 分，共 30 分）}

\needspace{8\baselineskip}

\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    如图意，一次函数 $y_1=x-2$ 的图像与 $y$ 轴交于点 $A$，一次函数 $y_2$ 的图像与 $y$ 轴交于点 $B(0,4)$，两函数图像交于点 $C$，且点 $C$ 的横坐标为 $3$。
\par\noindent (1) 求点 $A,C$ 的坐标及一次函数 $y_2$ 的解析式；
\par\noindent (2) 求 $\triangle ABC$ 的面积；
\par\noindent (3) 在坐标轴上，是否存在一点 $P$，使得 $S_{\triangle AOP}=2S_{\triangle BCP}$？若存在，直接写出点 $P$ 的坐标；若不存在，说明理由。

  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{第1题图}
  \input{\detokenize{build/diagram/jobs/practice-teacher-p1-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[70mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}(1) $A(0,-2)$，$C(3,1)$，$y_2=-x+4$；(2) $S_{\triangle ABC}=9$；(3) 存在，$P(3,0)$ 或 $P(6,0)$。\par\medskip
    \solstep{求点和函数}{由 $y_1=x-2$ 与 $y$ 轴相交，得 $A(0,-2)$。点 $C$ 的横坐标为 $3$，代入 $y_1=x-2$ 得 $C(3,1)$。直线 $y_2$ 过 $B(0,4)$ 和 $C(3,1)$，斜率为 $\dfrac{1-4}{3-0}=-1$，所以 $y_2=-x+4$。}
    \solstep{求面积}{$AB$ 在 $y$ 轴上，$AB=6$，点 $C$ 到 $y$ 轴的距离为 $3$，所以 $S_{\triangle ABC}=\dfrac12\cdot6\cdot3=9$。}
    \solstep{讨论 $P$ 在坐标轴上}{若 $P$ 在 $y$ 轴上，则 $A,O,P$ 共线，$S_{\triangle AOP}=0$，而 $S_{\triangle BCP}>0$，不符合。若 $P$ 在 $x$ 轴上，设 $P(t,0)$，则 $S_{\triangle AOP}=|t|$。}
    \solstep{列面积方程}{由 $B(0,4),C(3,1),P(t,0)$，得 $S_{\triangle BCP}=\dfrac12|3t-12|$。条件 $S_{\triangle AOP}=2S_{\triangle BCP}$ 化为 $|t|=|3t-12|$，解得 $t=3$ 或 $t=6$。故 $P(3,0)$ 或 $P(6,0)$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    已知一次函数 $y_1=kx+b$ 与反比例函数 $y_2=\dfrac{m}{x}$ 的图像交于点 $A,B$。已知：点 $A$ 的横坐标为 $3$，点 $B$ 的横坐标为 $-1$，一次函数图像与 $y$ 轴交于点 $C(0,-2)$。
\par\noindent (1) 求一次函数解析式、反比例函数解析式，以及点 $A,B$ 的坐标；
\par\noindent (2) 根据图像求不等式 $kx+b>\dfrac{m}{x}$ 的解集；
\par\noindent (3) 点 $P$ 在 $x$ 轴上，若 $S_{\triangle AOP}=S_{\triangle PAB}$，求点 $P$ 的坐标。

  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{第2题图}
  \input{\detokenize{build/diagram/jobs/practice-teacher-p2-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[72mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}(1) $y_1=x-2$，$y_2=\dfrac3x$，$A(3,1)$，$B(-1,-3)$；(2) $-1<x<0$ 或 $x>3$；(3) $P(\dfrac85,0)$ 或 $P(\dfrac83,0)$。\par\medskip
    \solstep{求函数和交点}{由 $C(0,-2)$ 得 $b=-2$。设 $A(3,3k-2)$，$B(-1,-k-2)$。因为 $A,B$ 都在 $y=\dfrac{m}{x}$ 上，所以 $3(3k-2)=(-1)(-k-2)$，解得 $k=1$。于是 $y_1=x-2$，$A(3,1)$，$B(-1,-3)$，$m=3$，$y_2=\dfrac3x$。}
    \solstep{图像解不等式}{$x-2>\dfrac3x$ 表示一次函数图像在反比例函数图像上方。断点为 $-1,0,3$，取 $(-1,0)$ 和 $(3,+\infty)$，所以解集为 $-1<x<0$ 或 $x>3$。}
    \solstep{设点并写面积}{设 $P(t,0)$。则 $S_{\triangle AOP}=\dfrac12|t|$。又 $A(3,1),B(-1,-3)$，所以 $S_{\triangle PAB}=\dfrac12|4t-8|=2|t-2|$。}
    \solstep{解面积方程并验算}{由 $S_{\triangle AOP}=S_{\triangle PAB}$，得 $\dfrac12|t|=2|t-2|$，即 $|t|=4|t-2|$。解得 $t=\dfrac85$ 或 $t=\dfrac83$。故 $P(\dfrac85,0)$ 或 $P(\dfrac83,0)$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    已知一次函数 $y_1=kx+b$ 与反比例函数 $y_2=\dfrac{m}{x}$ 的图像交于点 $A,B$。已知：$k=-1$，$m=-2$，点 $A$ 的横坐标为 $1$。
\par\noindent (1) 求一次函数解析式、反比例函数解析式，以及点 $A,B$ 的坐标；
\par\noindent (2) 根据图像求不等式 $kx+b>\dfrac{m}{x}$ 的解集；
\par\noindent (3) 点 $P$ 在反比例函数第四象限分支上，且在点 $A$ 的右侧。若 $S_{\triangle PAB}=3$，求点 $P$ 的坐标。

  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{第3题图}
  \input{\detokenize{build/diagram/jobs/practice-teacher-p3-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[76mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}(1) $y_1=-x-1$，$y_2=-\dfrac2x$，$A(1,-2)$，$B(-2,1)$；(2) $x<-2$ 或 $0<x<1$；(3) $P(2,-1)$。\par\medskip
    \solstep{求函数和交点}{由 $m=-2$ 且 $x_A=1$，得 $A(1,-2)$。又 $k=-1$，代入 $y=-x+b$ 得 $-1+b=-2$，所以 $b=-1$，$y_1=-x-1$。解 $-x-1=-\dfrac2x$，得 $x=1$ 或 $x=-2$，所以 $B(-2,1)$。}
    \solstep{图像解不等式}{$-x-1>-\dfrac2x$。两个交点横坐标为 $-2,1$，且 $x=0$ 不能取。由图像上下关系，直线在双曲线上方的区间为 $(-\infty,-2)$ 和 $(0,1)$。}
    \solstep{设点并找竖直距离}{点 $P$ 在第四象限分支且在 $A$ 右侧，设 $P(t,-\dfrac2t)$，其中 $t>1$。直线 $AB$ 为 $y=-x-1$，过 $P$ 作铅垂线交 $AB$ 于 $H$，则 $H(t,-t-1)$。}
    \solstep{列面积方程}{$PH=|(-t-1)-(-\dfrac2t)|=|t+1-\dfrac2t|$。因为 $t>1$，所以 $t+1-\dfrac2t>0$。又 $|x_A-x_B|=3$，所以 $S_{\triangle PAB}=\dfrac12\cdot3\cdot(t+1-\dfrac2t)=3$。}
    \solstep{求点并验算}{由 $\dfrac32(t+1-\dfrac2t)=3$，得 $t+1-\dfrac2t=2$，即 $t^2-t-2=0$。解得 $t=2$ 或 $t=-1$，结合 $t>1$，取 $t=2$，所以 $P(2,-1)$。代回面积为 $\dfrac12\cdot3\cdot2=3$。}
  \end{solutionblock}
\end{problem}
