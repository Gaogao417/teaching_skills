\begin{eduAuxItem}{思路导航}
先证明两个三角形相似$\;\to\;$再列对应边比例求边长\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题}
已知 $AO=6$，$OC=8$，$OD=9$，求 $OB$。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{对应边怎么看}{先看等角的顶点：$A\leftrightarrow D$，交点 $O\leftrightarrow O$，于是 $C\leftrightarrow B$；边由两个端点决定，所以 $AO\leftrightarrow DO$，$OC\leftrightarrow OB$。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{先证明两个三角形相似}{由对顶角相等，得 $\angle AOC=\angle DOB$。又 $\angle OAC=\angle ODB$，所以 $\triangle AOC\sim\triangle DOB$。}
      \eduExplainStep{2}{再列对应边比例求边长}{由对应边成比例，得 $\dfrac{AO}{DO}=\dfrac{OC}{OB}$，即 $\dfrac69=\dfrac8{OB}$，解得 $OB=12$。}
  \end{eduExplainRight}
\end{eduDualExplain}
