\eduSectionTitle{子母型：证明相似后求边长}

\begin{eduProblemBlock}[例题 1]
\noindent
\begin{minipage}[t]{\dimexpr\linewidth-70mm-6mm\relax}
\vspace{0pt}
如图，在 $\triangle ABC$ 中，$\angle ABD=\angle ACB$。已知 $AC=9$，$CD=5$。

（1）求 $AB$；

（2）若 $BC=12$，求 $BD$。

\end{minipage}\hfill
\begin{diagramcoltikz}{70mm}{先找出两个相似三角形，再判断 $CD$ 属不属于它们的边。}
\input{\detokenize{build/diagram/jobs/explanation-nested-prompt/rendered/prompt.fragment.tex}}
\end{diagramcoltikz}
\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
先证明两个三角形相似$\;\to\;$再列对应边比例求边长\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题}
（1）求 $AB$；（2）若 $BC=12$，求 $BD$。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{对应边怎么看}{先看等角的顶点：$A\leftrightarrow A$、$B\leftrightarrow C$，于是 $D\leftrightarrow B$；因此 $AB\leftrightarrow AC$、$AD\leftrightarrow AB$、$BD\leftrightarrow BC$。$CD$ 不在小三角形中，列比例前先换成 $AD=AC-CD$。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{先证明两个三角形相似}{由 $D\in AC$，得 $\angle BAD=\angle CAB$。又因为 $\angle ABD=\angle ACB$，所以 $\triangle ABD\sim\triangle ACB$。}
      \eduExplainStep{2}{再列对应边比例求边长}{先得 $AD=AC-CD=4$。由 $\dfrac{AC}{AB}=\dfrac{AB}{AD}$，得 $AB^2=9\times4$，所以 $AB=6$；再由 $\dfrac{BC}{BD}=\dfrac{AC}{AB}=\dfrac96$，得 $BD=8$。}
  \end{eduExplainRight}
\end{eduDualExplain}
