\eduSectionTitle{ASA：从第三顶点作高，让两个已知角共享高}


\begin{eduSolutionBox}{核心动作：两个角都以高为对边}
  \begingroup
  \setlength{\parskip}{2.5mm}
已知角在 $A,C$，夹边是 $AC$，所以从没有给已知角的顶点 $B$ 作 $BH\perp AC$。\par
在 Rt$\triangle ABH$ 中，$\tan A=BH/AH$；在 Rt$\triangle BCH$ 中，$\tan C=BH/CH$。\par
列式后先看同一条高 $BH$ 在两套比中各占几份，用最小公倍数统一份数，再用 $AH+CH=AC$ 定标。\par
  \endgroup
\end{eduSolutionBox}




\begin{eduProblemBlock}[例题]
\noindent
\begin{minipage}[t]{\dimexpr\linewidth-70mm-6mm\relax}
\vspace{0pt}
在 $\triangle ABC$ 中，已知 $\angle A$、$\angle C$ 及其夹边 $AC=25$，其中
\[
\tan A=\frac34,\qquad \tan C=\frac43.
\]
求 $BH,AB,BC$，并完整解出 $\triangle ABC$。

\end{minipage}\hfill

\end{diagramcoltikz}
\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
从未给已知角的顶点作高$\;\to\;$写两个以高为对边的正切比$\;\to\;$先统一高的份数$\;\to\;$用夹边总长定标$\;\to\;$分别解两个 Rt 三角形$\;\to\;$补出第三角并验算\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题}
求 $BH,AB,BC$，并完整解出 $\triangle ABC$。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{高从哪里作}{已知角在 $A,C$，所以从第三个顶点 $B$ 作高；这样两个已知角都能直接使用。}
        \eduSideHint{为什么不是 $BH=x$}{若设 $BH=x$，会得到 $AH=4x/3,CH=3x/4$；设 $BH=12x$ 可直接得到整数份数。}
        \eduSideHint{先看哪一条线}{两个正切比都以 $BH$ 为分子；先统一这条公共高的份数，再拼 $AC$。}
        \eduSideMistake{不要写反正切比}{对角 $A$，对边是 $BH$、邻边是 $AH$；对角 $C$，对边仍是 $BH$、邻边是 $CH$。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{从未给已知角的顶点作高}{过 $B$ 作 $BH\perp AC$ 于 $H$，得到 Rt$\triangle ABH$ 与 Rt$\triangle BCH$。}
      \begin{center}
        \begin{diagramcoltikz}{70mm}{从 $B$ 作高后，$A,C$ 分别留在两个 Rt 三角形中。}
        \input{\detokenize{build/diagram/jobs/explanation-asa-altitude/rendered/solution.fragment.tex}}
        \end{diagramcoltikz}
      \end{center}
      \eduExplainStep{2}{写两个以高为对边的正切比}{$BH:AH=3:4$，$BH:CH=4:3$。}
      \eduExplainStep{3}{先统一高的份数}{$BH$ 既是 3 的倍数又是 4 的倍数，设 $BH=12x$，则 $AH=16x,CH=9x$。}
      \eduExplainStep{4}{用夹边总长定标}{$AC=AH+CH=16x+9x=25x=25$，所以 $x=1$。}
      \eduExplainStep{5}{分别解两个 Rt 三角形}{$BH=12,AH=16,CH=9$；由勾股定理得 $AB=20,BC=15$。}
      \eduExplainStep{6}{补出第三角并验算}{$A\approx36.9^\circ,C\approx53.1^\circ$，故 $B=90^\circ$；且 $12^2+16^2=20^2$，$12^2+9^2=15^2$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduMistakeBox}
\eduSubTitle{易错提醒：先看高的份数}两个比中的 3 份高和 4 份高是同一条 $BH$，不能直接把两套份数相加。先统一成 12 份，再求两段底边。\end{eduMistakeBox}




\begin{eduSummaryBlock}{\eduIconPin}{ASA 做题顺序}
\begin{eduBulletList}
  \item 圈出两个已知角和夹边 $\rightarrow$ 从第三顶点向夹边作高。  \item 两个已知角都以高为对边 $\rightarrow$ 看高的份数 $\rightarrow$ 取最小公倍数。  \item 两段底边拼成夹边 $\rightarrow$ 定标 $\rightarrow$ 分别解两个 Rt 三角形。\end{eduBulletList}
\end{eduSummaryBlock}
