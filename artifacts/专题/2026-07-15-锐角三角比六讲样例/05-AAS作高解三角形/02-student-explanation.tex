\eduSectionTitle{AAS：从未给角的顶点作高，让公共高连接两个 Rt 三角形}


\begin{eduSolutionBox}{固定路线：圈出两个已知角，剩余顶点就是作高顶点}
  \begingroup
  \setlength{\parskip}{2.5mm}
\textbf{先辨型：}已知 $\angle A,\angle C$ 和非夹边 $AB$，这是 AAS，三角形唯一确定。\par
\textbf{再作高：}两个已知角顶点是 $A,C$，因此从唯一未给角的顶点 $B$ 作 $BH\perp AC$。\par
\textbf{先解第一个 Rt：}$\operatorname{Rt}\triangle ABH$ 含已知斜边 $AB$ 和已知锐角 $A$，先求公共高 $BH$ 与投影 $AH$。\par
\textbf{再解第二个 Rt：}把 $BH$ 带入 $\operatorname{Rt}\triangle BCH$，结合已知角 $C$ 求 $CH,BC$。\par
\textbf{最后还原：}$AC=AH+CH$，再用内角和求 $\angle B$。\par
\begin{center}
\begin{tabular}{c|c|c}
阶段 & 已知入口 & 求出的量 \\
\hline
$\operatorname{Rt}\triangle ABH$ & $AB$、$\angle A$ & $BH$、$AH$ \\
公共高传递 & $BH$ 同时属于两个 Rt & 把条件送入右侧 \\
$\operatorname{Rt}\triangle BCH$ & $BH$、$\angle C$ & $CH$、$BC$
\end{tabular}
\end{center}
\par
\textbf{边界：}若已知边改为两角夹边 $AC$，题型就变为 ASA；AAS 与 ASA 都唯一解，但第一步可用的已知边位置不同。\par
  \endgroup
\end{eduSolutionBox}




\begin{eduProblemBlock}[例题]
\noindent
\begin{minipage}[t]{\dimexpr\linewidth-70mm-6mm\relax}
\vspace{0pt}
在三角形 $ABC$ 中，已知 $\angle A=30^\circ$，$\angle C=45^\circ$，$AB=8$。
不使用正弦定理、余弦定理，从唯一未作为已知角顶点的点 $B$ 向 $AC$ 作高 $BH$，
求 $AC$、$BC$ 和 $\angle B$。

\end{minipage}\hfill

\begin{eduAuxItem}{思路导航}
确认 AAS 与作高顶点$\;\to\;$从 $B$ 向两角夹边 $AC$ 作高$\;\to\;$在 $\triangle ABH$ 中求公共高和左投影$\;\to\;$在 $\triangle BCH$ 中求右投影$\;\to\;$在 $\triangle BCH$ 中求 $BC$$\;\to\;$合并两段投影求 $AC$$\;\to\;$用内角和求 $\angle B$ 并验算\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题}
从未给角的顶点作高，依次解两个直角三角形，求 $AC$、$BC$ 和 $\angle B$。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{作高顶点怎么找}{先圈两个已知角 $A,C$；剩下的顶点 $B$ 就是作高顶点。不要从已知角顶点作高。}
        \eduSideHint{为什么先解左侧}{$\triangle ABH$ 同时有已知斜边 $AB=8$ 和已知锐角 $A=30^\circ$，右侧此时只有角 $C$。}
        \eduSideHint{公共高有什么用}{左侧求出的 $BH=4$，正好是右侧相对于 $\angle C$ 的对边，于是第二个直角三角形也能解。}
        \eduSideHint{$AC$ 为什么用加法}{$H$ 在线段 $AC$ 上，且 $A,H,C$ 依次共线，所以 $AC=AH+CH$。}
        \eduSideNote{AAS 会有两解吗}{不会。两个角先确定三角形形状，再由一条边确定大小，因此 AAS 唯一解。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{确认 AAS 与作高顶点}{已知 $\angle A,\angle C$ 和非夹边 $AB$；唯一未给角的顶点为 $B$。}
      \eduExplainStep{2}{从 $B$ 向两角夹边 $AC$ 作高}{作 $BH\perp AC$ 于 $H$，得到 $\operatorname{Rt}\triangle ABH$ 与 $\operatorname{Rt}\triangle BCH$。}
      \begin{center}
        \begin{diagramcoltikz}{70mm}{公共高 $BH$ 把两个已知角分别放进两个直角三角形。}
        \input{\detokenize{build/diagram/jobs/explanation-aas-altitude-solution/rendered/solution.fragment.tex}}
        \end{diagramcoltikz}
      \end{center}
      \eduExplainStep{3}{在 $\triangle ABH$ 中求公共高和左投影}{$BH=8\sin30^\circ=4$，$AH=8\cos30^\circ=4\sqrt3$。}
      \eduExplainStep{4}{在 $\triangle BCH$ 中求右投影}{$\tan45^\circ=\dfrac{BH}{CH}=1$，故 $CH=4$。}
      \eduExplainStep{5}{在 $\triangle BCH$ 中求 $BC$}{$BC=\dfrac{BH}{\sin45^\circ}=\dfrac{4}{\sqrt2/2}=4\sqrt2$。}
      \eduExplainStep{6}{合并两段投影求 $AC$}{$AC=AH+CH=4\sqrt3+4$。}
      \eduExplainStep{7}{用内角和求 $\angle B$ 并验算}{$\angle B=180^\circ-30^\circ-45^\circ=105^\circ$；最大边 $AC$ 对应最大角 $B$。}
    \vspace{1mm}
    {\color{edu-blue}\bfseries\small 检查}\par\vspace{1mm}
    \begin{eduBulletList}
      \item $30^\circ+45^\circ+105^\circ=180^\circ$。      \item $AC=4\sqrt3+4$ 为最大边，对应最大角 $\angle B=105^\circ$。      \item $AB=8>BC=4\sqrt2$，对应 $\angle C=45^\circ>\angle A=30^\circ$。    \end{eduBulletList}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduMistakeBox}
\eduSubTitle{易错提醒：AAS 的已知边不是两角夹边}两个已知角是 $A,C$，它们的夹边是 $AC$；题目给的是 $AB$，所以这是 AAS。
作高时先找唯一未给角的顶点 $B$，再向两角夹边 $AC$ 作垂线。若把已知边换成
$AC$，题型就成为 ASA，不能继续照搬“已知斜边 $AB$ 先解左侧”的入口。
\end{eduMistakeBox}




\begin{eduSummaryBlock}{\eduIconPin}{AAS 作高的五步口令}
\begin{eduBulletList}
  \item 辨：两角的夹边是哪条，已知边是否为非夹边。  \item 作：从唯一未给角的顶点向两角夹边作高。  \item 一：先解同时含已知边和已知角的第一个 $\operatorname{Rt}\triangle$。  \item 传：把求出的公共高带入第二个 $\operatorname{Rt}\triangle$。  \item 二：解第二个 Rt，合并投影还原原边，再用内角和验算。\end{eduBulletList}
\end{eduSummaryBlock}
