\eduSectionTitle{SSS 已知三边长，解三角形}

\begin{eduAuxItem}{小贴士}
“解三角形”就是利用一个三角形中已知的三个条件，求出这个三角形还不知道的边、角或高。
\end{eduAuxItem}

\eduSectionTitle{先按锐角三角形来解}

\begin{eduProblemBlock}[例题 1]
\noindent
\begin{minipage}[t]{\dimexpr\linewidth-70mm-6mm\relax}
\vspace{0pt}
在 $\triangle ABC$ 中，$AB=13$，$AC=15$，$BC=14$，三个角均为锐角。
过 $A$ 作 $AD\perp BC$ 于 $D$。求高 $AD$ 及 $\tan B$、 $\tan C$ 。

\end{minipage}\hfill
\begin{diagramcoltikz}{70mm}{先看三条已知边；高要由我们自己作出。}
\input{\detokenize{build/diagram/jobs/explanation-sss_acute-prompt/rendered/prompt.fragment.tex}}
\end{diagramcoltikz}
\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
作高，拆成两个 Rt 三角形$\;\to\;$设底边分段，不设高$\;\to\;$双勾股消去同一个 $AD^2$$\;\to\;$回到一个 Rt 三角形求高$\;\to\;$在两个 Rt 三角形中求锐角\end{eduAuxItem}





\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{为什么设 $BD=x$}{底边另一段能立刻写成 $14-x$；若设高为 $x$，还要另设底边分段。}
        \eduSideHint{双勾股在消什么}{两个 Rt 三角形共用 $AD$，让两个 $AD^2$ 表达式相等即可消去高。}
        \eduSideMistake{三角比写在哪里}{必须写在 Rt$\triangle ABD$ 或 Rt$\triangle ACD$ 中，不能直接把原三角形当直角三角形。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{作高，拆成两个 Rt 三角形}{过 $A$ 作 $AD\perp BC$ 于 $D$，得到 Rt$\triangle ABD$ 与 Rt$\triangle ACD$。}
      \begin{center}
        \begin{diagramcoltikz}{70mm}{作高后，底边 $BC$ 被分成两段。}
        \input{\detokenize{build/diagram/jobs/explanation-sss_acute-altitude/rendered/solution.fragment.tex}}
        \end{diagramcoltikz}
      \end{center}
      \eduExplainStep{2}{设底边分段为 $x$，不设高为 $x$}{设 $BD=x$。先按锐角三角形的图来处理，写出 $DC=14-x$。}
      \eduExplainStep{3}{双勾股消去同一个 $AD^2$}{$13^2-x^2=AD^2=15^2-(14-x)^2$，解得 $x=5$。}
      \eduExplainStep{4}{回到一个 Rt 三角形求高}{$BD=5,DC=9$，故 $AD=\sqrt{13^2-5^2}=12$。}
      \eduExplainStep{5}{在两个 Rt 三角形中求锐角}{$\tan B=\dfrac{12}{5}$，$\tan C=\dfrac{12}{9}=\dfrac43$。}
  \end{eduExplainRight}
\end{eduDualExplain}

\eduSectionTitle{若 $x$ 是负数：原三角形是钝角三角形}

\begin{eduProblemBlock}[例题 2]
\noindent
\begin{minipage}[t]{\dimexpr\linewidth-70mm-6mm\relax}
\vspace{0pt}
在 $\triangle ABC$ 中，$AB=13$，$AC=20$，$BC=11$。
过 $A$ 作 $AD\perp BC$ 所在直线于 $D$。求 $AD$ 及 $\triangle ABC$ 的三个角
（角度保留一位小数）。

\end{minipage}\hfill
\begin{diagramcoltikz}{70mm}{先只使用三条已知边，不需要预先判断三角形的形状。}
\input{\detokenize{build/diagram/jobs/explanation-sss_obtuse-prompt/rendered/prompt.fragment.tex}}
\end{diagramcoltikz}
\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
先当成锐角三角形设 $BD=x$、$DC=11-x$$\;\to\;$双勾股求 $x$$\;\to\;$若 $x<0$，判断原三角形是钝角三角形$\;\to\;$用 $|x|$ 求高和各角\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 2}
求高 $AD$ 及 $\triangle ABC$ 的三个角。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{为什么仍然写 $11-x$}{这是一套统一的设法：先默认三角形是锐角三角形，让计算结果告诉我们垂足的真实位置。}
        \eduSideHint{$x=-5$ 表示什么}{负号说明实际的垂足在假设方向的反方向；$BD$ 的实际长度是 $|x|=5$。}
        \eduSideHint{哪一个角能直接用三角比}{$\angle ABD$ 是 Rt$\triangle ABD$ 中的锐角；原 $\angle ABC$ 是它的补角。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{先按锐角三角形设元}{设 $BD=x$，先按锐角三角形的图写出 $DC=11-x$。}
      \eduExplainStep{2}{双勾股求 $x$}{$13^2-x^2=AD^2=20^2-(11-x)^2$，解得 $x=-5$。}
      \eduExplainStep{3}{用负号判断三角形}{解得 $x<0$，说明垂足在 $B$ 的外侧，因此 $\angle B$ 是钝角，$\triangle ABC$ 是钝角三角形。}
      \begin{center}
        \begin{diagramcoltikz}{70mm}{$x=-5$ 告诉我们：垂足实际落在 $B$ 的外侧。}
        \input{\detokenize{build/diagram/jobs/explanation-sss_obtuse-altitude/rendered/solution.fragment.tex}}
        \end{diagramcoltikz}
      \end{center}
      \eduExplainStep{4}{用实际长度求高}{$BD=|x|=5$，$DC=11-(-5)=16$，故 $AD=\sqrt{13^2-5^2}=12$。}
      \eduExplainStep{5}{先求锐角，再还原钝角}{$\tan\angle ABD=\dfrac{12}{5}$，故 $B=180^\circ-\angle ABD\approx112.6^\circ$；又 $\tan C=\dfrac{12}{16}=\dfrac34$，故 $C\approx36.9^\circ$，$A\approx30.5^\circ$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduSummaryBlock}{\eduIconPin}{统一设法}
\begin{eduBulletList}
  \item 不必先判断锐角或钝角；统一先设 $BD=x$，并写出 $DC=BC-x$。
  \item $0<x<BC$ 时，垂足在线段 $BC$ 上，原三角形是锐角三角形。
  \item $x<0$ 时，垂足在 $B$ 的外侧，$\angle B$ 是钝角；$x>BC$ 时，垂足在 $C$ 的外侧，$\angle C$ 是钝角。
  \item $x=0$ 或 $x=BC$ 时，垂足分别与 $B$、$C$ 重合，原三角形是直角三角形。
\end{eduBulletList}
\eduSubTitle{想一想}以下三题都过 $A$ 作 $AD\perp BC$ 所在直线于 $D$。设 $BD=x$，只需求出 $x$，再判断垂足位置和三角形类型。
\begin{eduBulletList}
  \item $AB=13,\ AC=15,\ BC=14$。
  \item $AB=13,\ AC=20,\ BC=11$。
  \item $AB=20,\ AC=13,\ BC=11$。
\end{eduBulletList}
\end{eduSummaryBlock}
