\eduSectionTitle{SAS：已知两边一角（夹角），解三角形}






\begin{eduProblemBlock}[例题]
\noindent
\begin{minipage}[t]{\dimexpr\linewidth-70mm-6mm\relax}
\vspace{0pt}
在三角形 $ABC$ 中，已知 $AB=10$，$AC=13$，夹角 $\angle A=30^\circ$。求边 $BC$ 以及
$\tan \angle C$。

\end{minipage}\hfill
\begin{diagramcoltikz}{70mm}{先圈出两条已知边和它们的夹角，再决定从哪个顶点作高。}
\input{\detokenize{build/diagram/jobs/explanation-sas-prompt/rendered/prompt.fragment.tex}}
\end{diagramcoltikz}
\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
确认标准 SAS 与唯一解$\;\to\;$从非已知角顶点 $B$ 作高$\;\to\;$在 $\triangle ABH$ 中求高和投影$\;\to\;$判断点序并求 $CH$$\;\to\;$在 $\triangle BCH$ 中求 $BC$ 和 $\tan C$\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题}
作高并依次解两个直角三角形，求 $BC$ 和 $\tan C$。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{为什么不从 $A$ 作高}{已知角就在 $A$。若从 $A$ 作高，已知角会被分开，第一个直角三角形不能直接用“斜边加锐角”求解。}
        \eduSideHint{先解哪一个 Rt}{$\triangle ABH$ 同时含已知斜边 $AB=10$ 和已知锐角 $A=30^\circ$，所以它必须先解。}
        \eduSideHint{为什么先比较 $AH$ 与 $AC$}{只有确认 $H$ 在线段 $AC$ 上，才能写 $CH=AC-AH$；垂足在延长线上时应改用线段和。}
        \eduSideNote{这里会有两解吗}{不会。标准 SAS 唯一确定三角形；SSA 才需要讨论无解、一解或两解。}
        \eduSideHint{第二个 Rt 得到了什么}{$BH$ 是公共高，$CH$ 来自线段差；两条直角边齐全后，就能求 $BC$ 和 $\tan C$。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{确认标准 SAS 与唯一解}{$AB,AC$ 及其夹角 $\angle A$ 已知，因此三角形由 SAS 唯一确定。}
      \eduExplainStep{2}{从非已知角顶点 $B$ 作高}{作 $BH\perp AC$ 于 $H$，得到 $\operatorname{Rt}\triangle ABH$ 与 $\operatorname{Rt}\triangle BCH$。}
      \begin{center}
        \begin{diagramcoltikz}{70mm}{从 $B$ 向 $AC$ 作高后，先解左侧的 $\operatorname{Rt}\triangle ABH$。}
        \input{\detokenize{build/diagram/jobs/explanation-sas-altitude-solution/rendered/solution.fragment.tex}}
        \end{diagramcoltikz}
      \end{center}
      \eduExplainStep{3}{在 $\triangle ABH$ 中求高和投影}{$BH=10\sin30^\circ=5$，$AH=10\cos30^\circ=5\sqrt3$。}
      \eduExplainStep{4}{求 $CH$}{ $H$ 在线段 $AC$ 上，$CH=13-5\sqrt3$。}
      \eduExplainStep{5}{在 $\triangle BCH$ 中求 $BC$}{$BC=\sqrt{5^2+(13-5\sqrt3)^2}=\sqrt{269-130\sqrt3}。$}
      \eduExplainStep{6}{在 $\triangle BCH$ 中求 $\tan C$}{$\tan C=\dfrac{5}{13-5\sqrt3}=\dfrac{5(13+5\sqrt3)}{13^2-(5\sqrt3)^2}=\dfrac{65+25\sqrt3}{94}$。}


  \end{eduExplainRight}
\end{eduDualExplain}
