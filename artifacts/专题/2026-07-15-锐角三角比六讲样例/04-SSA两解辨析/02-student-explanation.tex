\eduSectionTitle{SSA 已知两边一角（非夹角），解三角形}






\begin{eduProblemBlock}[例题]
\noindent
\begin{minipage}[t]{\dimexpr\linewidth-70mm-6mm\relax}
\vspace{0pt}
在 $\triangle ABC$ 中，已知 $\angle A=30^\circ$，$AB=8$，$BC=5$。
求$AC$。

\end{minipage}\hfill
\begin{diagramcoltikz}{70mm}{点 $C$ 既要在 $30^\circ$ 射线上，又要满足到 $B$ 的距离为 $5$；先判断这样的位置是否唯一。}
\input{\detokenize{build/diagram/jobs/explanation-ssa-prompt/rendered/prompt.fragment.tex}}
\end{diagramcoltikz}
\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
识别 SSA，不先假定图形唯一$\;\to\;$从非已知角顶点 $B$ 作高$\;\to\;$先比较大小，确认有两个解$\;\to\;$在第二个直角三角形中求 $HC$$\;\to\;$近点分支：垂足在线段 $AC$ 的延长线上$\;\to\;$远点分支：垂足在线段 $AC$ 内$\;\to\;$检查两个候选三角形\end{eduAuxItem}





\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{为什么从 $B$ 作高}{$B$ 不是已知角顶点；由 $AB$ 和 $\angle A$ 可直接求出高 $BH$ 与投影 $AH$。}
        \eduSideHint{为什么先比较 $4,5,8$}{$4<5$ 表示定长 $BC$ 能跨过垂足两侧；$5<8$ 保证靠近 $A$ 的候选点仍留在射线上。}
        \eduSideHint{如果 $BC=9$ 呢}{这时 $AC$ 只有一个解：以 $B$ 为圆心、$BC$ 为半径作圆，它与射线 $AH$ 只有一个交点，因此只对应一个三角形。}
        \eduSideHint{两解从哪里来}{两解不是代数中的符号游戏，而是 $C_1,C_2$ 位于 $H$ 两侧的两个真实位置。}
        
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{识别 SSA，不先假定图形唯一}{已知 $AB,BC$ 及 $BC$ 的对角 $\angle A$；$\angle A$ 不是两条已知边的夹角，因此属于 SSA。}
      \eduExplainStep{2}{从非已知角顶点 $B$ 作高}{向 $AC$ 所在直线作 $BH\perp AC$ 于 $H$。在 $\operatorname{Rt}\triangle ABH$ 中，$BH=8\sin30^\circ=4$，$AH=8\cos30^\circ=4\sqrt3$。}
      \begin{center}
        \begin{diagramcoltikz}{70mm}{两个候选点位于垂足 $H$ 两侧，点序分别给出线段差与线段和。}
        \input{\detokenize{build/diagram/jobs/explanation-ssa-altitude/rendered/solution.fragment.tex}}
        \end{diagramcoltikz}
      \end{center}
      \eduExplainStep{3}{先比较大小，确认有两个解}{因为 $BH=4<BC=5<AB=8$，即 $h<a<c$，所以点 $C$ 在垂足 $H$ 两侧各有一个合法位置。}
      \eduExplainStep{4}{在第二个直角三角形中求 $HC$}{在 $\operatorname{Rt}\triangle BCH$ 中，由勾股定理得 $HC=\sqrt{BC^2-BH^2}=\sqrt{5^2-4^2}=3$。}
      \eduExplainStep{5}{近点分支：垂足在线段 $AC$ 的延长线上}{当点序为 $A-C_1-H$ 时，$AC_1=AH-C_1H=4\sqrt3-3$。}
      \eduExplainStep{6}{远点分支：垂足在线段 $AC$ 内}{当点序为 $A-H-C_2$ 时，$AC_2=AH+HC_2=4\sqrt3+3$。}
      \eduExplainStep{7}{检查两个候选三角形}{两个 $AC$ 均为正，且分别满足三角形不等式；故两种点序都对应合法三角形。}
    \vspace{1mm}
    {\color{edu-blue}\bfseries\small 结论}\par\vspace{1mm}
    \begin{eduBulletList}
      \item $\boxed{AC=4\sqrt3-3\quad\text{或}\quad AC=4\sqrt3+3}$，共有两个三角形。    \end{eduBulletList}
  \end{eduExplainRight}
\end{eduDualExplain}
