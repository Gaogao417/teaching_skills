\section*{比例辅助线边比练习}

\section*{一、核心练习}

\needspace{8\baselineskip}

\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    如图，在 $\triangle ABC$ 中，点 $D$ 在边 $AB$ 上，点 $E$ 在边 $AC$ 上，$CD$ 与 $BE$ 相交于点 $P$。
已知 $AD:DB=1:2$，$AE:EC=1:2$，求 $CP:PD$。

    \begin{hintbox}
\textbf{提示：}目标在 $CD$ 上，先想 $CP:PD=(AB:DB)\times(CE:AE)$。\end{hintbox}

  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{第 1 题图}
  \input{\detokenize{build/diagram/jobs/practice-teacher-q1-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[45mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$CP:PD=3:1$。\par\medskip
    \solstep{选公式}{目标在 $CD$ 上，用 $CP:PD=(AB:DB)\times(CE:AE)$。}
    \solstep{转整段比}{$AD:DB=1:2$，所以 $AB:DB=(AD+DB):DB=3:2$。}
    \solstep{转反向比}{$AE:EC=1:2$，所以 $CE:AE=2:1$。}
    \solstep{相乘}{$CP:PD=(3:2)\times(2:1)=3:1$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    如图，在 $\triangle ABC$ 中，点 $D$ 在边 $AB$ 上，点 $E$ 在边 $AC$ 上，$CD$ 与 $BE$ 相交于点 $P$。
已知 $AD:DB=\frac{4}{3}:1$，$AE:EC=\frac{1}{2}:1$，求 $BP:PE$。

    \begin{hintbox}
\textbf{提示：}目标在 $BE$ 上，先想 $BP:PE=(AC:CE)\times(BD:AD)$。\end{hintbox}

  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{第 2 题图}
  \input{\detokenize{build/diagram/jobs/practice-teacher-q2-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[45mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$BP:PE=9:8$。\par\medskip
    \solstep{选公式}{目标在 $BE$ 上，用 $BP:PE=(AC:CE)\times(BD:AD)$。}
    \solstep{转整段比}{$AE:EC=\frac12:1$，所以 $AC:CE=(AE+EC):CE=\frac32:1=3:2$。}
    \solstep{转反向比}{$AD:DB=\frac43:1$，所以 $BD:AD=1:\frac43=3:4$。}
    \solstep{相乘}{$BP:PE=(3:2)\times(3:4)=9:8$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    如图，在 $\triangle ABC$ 中，点 $D$ 在边 $AB$ 上，点 $E$ 在边 $AC$ 上，$CD$ 与 $BE$ 相交于点 $P$。
已知 $AE:EC=\sqrt{2}:1$，$CP:PD=2\sqrt{2}:1$，求 $AD:DB$。

    \begin{hintbox}
\textbf{提示：}反过来用 $CP:PD=(AB:DB)\times(CE:AE)$，先求 $AB:DB$。\end{hintbox}

  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{第 3 题图}
  \input{\detokenize{build/diagram/jobs/practice-teacher-q3-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[45mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$AD:DB=3:1$。\par\medskip
    \solstep{反用比例链}{$CP:PD=(AB:DB)\times(CE:AE)$。}
    \solstep{代入已知}{$AE:EC=\sqrt2:1$，所以 $CE:AE=1:\sqrt2$；又 $CP:PD=2\sqrt2:1$。}
    \solstep{求整段比}{因为 $2\sqrt2=(AB:DB)\cdot\frac{1}{\sqrt2}$，所以 $AB:DB=4:1$。}
    \solstep{转回分段比}{$AD:DB=(AB-DB):DB=3:1$。}
  \end{solutionblock}
\end{problem}
