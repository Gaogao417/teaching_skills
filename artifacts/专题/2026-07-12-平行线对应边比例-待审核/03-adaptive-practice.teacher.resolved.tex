\section*{一、A 字型（3 题）}

\needspace{8\baselineskip}

\begin{problem}[points=10]
    如图，直线 $AC$ 与 $BD$ 相交于点 $P$，点序分别为 $P-A-C$、$P-B-D$，且 $AB\parallel CD$。
已知 $PA:AC=1:2$，求 $AB:CD$。


  \answerarea[38mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$AB:CD=1:3$\par\medskip
    \solstep{根据点序补出整段}{由 $P-A-C$，$PC=PA+AC$，所以 $PA:PC=1:(1+2)=1:3$。}
    \solstep{写出相似与对应边}{由 $AB\parallel CD$，$△PAB\sim△PCD$，故 $AB:CD=PA:PC$。}
    \solstep{得出结果}{$AB:CD=1:3$。}
  \end{solutionblock}
\end{problem}
