\eduSectionTitle{先围三角形，再决定平行线和交点}


\begin{eduSolutionBox}{核心动作：已知和所求共同决定辅助线}
  \begingroup
  \setlength{\parskip}{2.5mm}
题目中的四条比例线是 $AC$ 上的 $AE:EC$、$AD$ 上的 $AP:PD$、$BE$ 上的 $BP:PE$、$BC$ 上的 $BD:DC$。\par
先找两条已知比和所求比所在的三条直线；它们围成的三角形，决定平行线从哪个顶点作出。\par
过这个三角形的一个顶点作对边的平行线；再连接平行线外的两个点，让这条连线与新平行线相交。\par
最后从完整图中找出两组 A/8 字型相似三角形，用共同的中间线段接比例。\par
  \endgroup
\end{eduSolutionBox}




\begin{eduProblemBlock}[例题]
\noindent
\begin{minipage}[t]{\dimexpr\linewidth-70mm-6mm\relax}
\vspace{0pt}
如图，在 $\triangle ABC$ 中，点 $D$ 在线段 $BC$ 上，点 $E$ 在线段 $AC$ 上，
线段 $AD$ 与 $BE$ 相交于点 $P$。已知
\[
AE:EC=1:2,\qquad BD:DC=2:1,
\]
求 $AP:PD$。

\end{minipage}\hfill

\begin{eduAuxItem}{思路导航}
找到三条比例线围成的三角形$\;\to\;$过顶点 $C$ 作对边的平行线$\;\to\;$用 $E$ 处的相似三角形读第一个比例$\;\to\;$用 $B$ 处的相似三角形读第二个比例$\;\to\;$统一中间线段 $CF$，写出所求比\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题}
求 $AP:PD$。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{先围哪一个三角形}{先看两条已知比和所求比分别在哪三条直线上；本题是 $AC,BC,AD$，所以围 $\triangle ACD$。}
        \eduSideHint{为什么交点不能随便选}{先确定平行线外的两个点 $B,E$，连接 $BE$ 后，新交点 $F$ 才由图形关系确定。}
        \eduSideMistake{不要漏掉整段 $BC$}{$BD:DC=2:1$ 先变成 $BD:BC=2:3$，不能直接写成 $BD:BC=2:1$。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{找到三条比例线围成的三角形}{$AE:EC$ 在 $AC$ 上，$BD:DC$ 在 $BC$ 上，$AP:PD$ 在 $AD$ 上；所以先围 $\triangle ACD$。}
      \eduExplainStep{2}{过顶点 $C$ 作对边的平行线}{过 $C$ 作 $CF\parallel AD$。点 $B,E$ 在这组平行线外，连接 $BE$，令它与 $CF$ 交于 $F$。}
      \begin{center}
        \begin{diagramcoltikz}{70mm}{围出 $\triangle ACD$ 后，过顶点 $C$ 作 $AD$ 的平行线。}
        \input{\detokenize{build/diagram/jobs/explanation-example-helper/rendered/solution.fragment.tex}}
        \end{diagramcoltikz}
      \end{center}
      \eduExplainStep{3}{用 $E$ 处的相似三角形读第一个比例}{由 $A,E,C$ 共线、$P,E,F$ 共线、$AP\parallel CF$，得 $\triangle EAP\sim\triangle ECF$，所以 $AP:CF=AE:EC=1:2$。}
      \begin{center}
        \begin{diagramcoltikz}{70mm}{先在 $E$ 处找出第一组相似三角形。}
        \input{\detokenize{build/diagram/jobs/explanation-example-similar-e/rendered/solution.fragment.tex}}
        \end{diagramcoltikz}
      \end{center}
      \eduExplainStep{4}{用 $B$ 处的相似三角形读第二个比例}{由 $B,D,C$ 共线、$B,P,F$ 共线、$DP\parallel CF$，得 $\triangle BDP\sim\triangle BCF$。又 $BD:BC=2:3$，所以 $DP:CF=2:3$。}
      \begin{center}
        \begin{diagramcoltikz}{70mm}{再在 $B$ 处找出第二组相似三角形。}
        \input{\detokenize{build/diagram/jobs/explanation-example-similar-b/rendered/solution.fragment.tex}}
        \end{diagramcoltikz}
      \end{center}
      \eduExplainStep{5}{统一中间线段 $CF$，写出所求比}{由 $AP:CF=1:2$、$DP:CF=2:3$，取 $CF=6$ 份，得 $AP=3$ 份、$DP=4$ 份，所以 $AP:PD=3:4$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduMistakeBox}
\eduSubTitle{易错提醒：不是先背一条辅助线}先由“两条已知比和一个所求比”围三角形，再确定顶点、对边和平行线外两点的连线；不能先背一条固定辅助线，也不能随便选交点。\end{eduMistakeBox}




\begin{eduSummaryBlock}{\eduIconPin}{做题顺序}
\begin{eduBulletList}
  \item 找三条比例线 $\rightarrow$ 围三角形 $\rightarrow$ 过顶点作对边平行线。  \item 找平行线外的两个点并连接 $\rightarrow$ 确定交点 $\rightarrow$ 找两组 A/8 字型。  \item 把两组比例中的中间线段统一份数，再求所需的分段比。\end{eduBulletList}
\end{eduSummaryBlock}
