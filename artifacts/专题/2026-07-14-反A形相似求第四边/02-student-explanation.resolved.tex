\eduSectionTitle{反 A 形：证明相似后求第四边}

\begin{eduProblemBlock}[例题]
\noindent
\begin{minipage}[t]{\dimexpr\linewidth-70mm-6mm\relax}
\vspace{0pt}
如图，$\angle PAB=\angle PDC$。已知 $PA=6$，$PB=8$，$PD=9$，求 $PC$。

\end{minipage}\hfill

\begin{eduAuxItem}{思路导航}
先证明两个三角形相似$\;\to\;$再列对应边比例求边长\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题}
求 $PC$。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{对应边怎么看}{先看等角的顶点：$A\leftrightarrow D$，公共顶点 $P\leftrightarrow P$，于是 $B\leftrightarrow C$；边由两个端点决定，所以 $PA\leftrightarrow PD$，$PB\leftrightarrow PC$。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{先证明两个三角形相似}{由共线关系，得 $\angle APB=\angle DPC$。又 $\angle PAB=\angle PDC$，所以 $\triangle PAB\sim\triangle PDC$。}
      \eduExplainStep{2}{再列对应边比例求边长}{由对应边成比例，得 $\dfrac{PA}{PD}=\dfrac{PB}{PC}$，即 $\dfrac69=\dfrac8{PC}$，解得 $PC=12$。}
  \end{eduExplainRight}
\end{eduDualExplain}
