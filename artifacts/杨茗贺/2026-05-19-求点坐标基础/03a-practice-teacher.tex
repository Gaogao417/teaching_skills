\section*{一、选择题（每题 12 分，共 48 分）}



\begin{keyideabox}
  \textbf{核心思想：直线与坐标轴的交点}
  求直线 $y=kx+b$ 与 $x$ 轴交点：令 $y=0$，解方程 $kx+b=0$，交点为 $\left(-\frac{b}{k}, 0\right)$。\\ 求与 $y$ 轴交点：令 $x=0$，得 $y=b$，交点为 $(0, b)$。
\par\small 坐标轴上的点特征：x 轴上的点纵坐标为 0，y 轴上的点横坐标为 0。\end{keyideabox}



\begin{question}[points=12]
  直线 $y = -3x + 9$ 与 $x$ 轴的交点为
  \begin{choices}
    \item $(3, 0)$
    \item $(0, 3)$
    \item $(-3, 0)$
    \item $(0, 9)$
  \end{choices}
  \paren[A]
  \begin{solutionblock}
    令 $y=0$：$0=-3x+9$，$x=3$。交点为 $(3, 0)$。
  \end{solutionblock}
\end{question}



\begin{question}[points=12]
  直线 $y = kx + 4$ 经过点 $(2, 0)$，则 $k$ 的值为
  \begin{choices}
    \item $2$
    \item $-2$
    \item $4$
    \item $-4$
  \end{choices}
  \paren[B]
  \begin{solutionblock}
    代入 $(2, 0)$：$0=2k+4$，$k=-2$。
  \end{solutionblock}
\end{question}



\begin{question}[points=12]
  直线 $y = 4x - 8$ 与 $y$ 轴的交点为
  \begin{choices}
    \item $(2, 0)$
    \item $(-2, 0)$
    \item $(0, 8)$
    \item $(0, -8)$
  \end{choices}
  \paren[D]
  \begin{solutionblock}
    令 $x=0$：$y=4 \times 0 - 8 = -8$。交点为 $(0, -8)$。
  \end{solutionblock}
\end{question}



\begin{question}[points=12]
  直线 $y = x + 1$ 与 $y = -2x + 7$ 的交点坐标为
  \begin{choices}
    \item $(1, 2)$
    \item $(2, 3)$
    \item $(3, 4)$
    \item $(2, 1)$
  \end{choices}
  \paren[B]
  \begin{solutionblock}
    令 $x+1=-2x+7$，$3x=6$，$x=2$。$y=2+1=3$。交点 $(2, 3)$。
  \end{solutionblock}
\end{question}


\section*{二、填空题（每题 12 分，共 12 分）}



\begin{question}[points=12]
  直线 $y = kx - 2$ 经过点 $(-3, 4)$，则 $k$ = \rule{3cm}{0.4pt}。
  \fillin
  \paren[$-2$]
  \begin{solutionblock}
    代入 $(-3, 4)$：$4 = -3k - 2$，$-3k = 6$，$k = -2$。
  \end{solutionblock}
\end{question}

\needspace{8\baselineskip}

\begin{problem}[points=20]
  直线 $y = -3x + 12$ 与 $x$ 轴、$y$ 轴分别交于点 $A$、$B$。

求 $A$、$B$ 的坐标。
  \answerarea[50mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$A(4, 0)$，$B(0, 12)$\par\medskip
    \solstep{求 $B$（$y$ 轴交点）}{令 $x = 0$：$y = -3 \times 0 + 12 = 12$，$B(0, 12)$。}
    \solstep{求 $A$（$x$ 轴交点）}{令 $y = 0$：$0 = -3x + 12$，$3x = 12$，$x = 4$，$A(4, 0)$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=20]
  直线 $l_1$：$y = 3x - 2$ 与直线 $l_2$：$y = -x + 6$。

求 $l_1$ 与 $l_2$ 的交点坐标。
  \answerarea[50mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$(2, 4)$\par\medskip
    \solstep{联立方程}{$3x - 2 = -x + 6$}
    \solstep{解 $x$}{$4x = 8$，$x = 2$}
    \solstep{回代求 $y$}{代入 $l_2$：$y = -2 + 6 = 4$。交点为 $(2, 4)$。}
  \end{solutionblock}
\end{problem}
