\section*{一、选择题（每题 12 分，共 48 分）}



\begin{question}[points=12]
  直线 $y = -2x + 8$ 与坐标轴的交点分别为
  \begin{choices}
    \item $(4, 0)$ 和 $(0, 8)$
    \item $(-4, 0)$ 和 $(0, 8)$
    \item $(4, 0)$ 和 $(0, -8)$
    \item $(-4, 0)$ 和 $(0, -8)$
  \end{choices}
  \paren[A]
  \begin{solutionblock}
    $x$ 轴交点：$0=-2x+8$，$x=4$，$(4,0)$。$y$ 轴交点：$y=8$，$(0,8)$。
  \end{solutionblock}
\end{question}



\begin{question}[points=12]
  直线 $y = kx + 3$ 经过点 $(1, 7)$，则 $k$ 的值为
  \begin{choices}
    \item $4$
    \item $-4$
    \item $10$
    \item $-10$
  \end{choices}
  \paren[A]
  \begin{solutionblock}
    代入 $(1, 7)$：$7=k+3$，$k=4$。
  \end{solutionblock}
\end{question}



\begin{question}[points=12]
  直线 $y = 5x - 10$ 与 $y$ 轴的交点为
  \begin{choices}
    \item $(2, 0)$
    \item $(-2, 0)$
    \item $(0, 10)$
    \item $(0, -10)$
  \end{choices}
  \paren[D]
  \begin{solutionblock}
    令 $x=0$：$y=5 \times 0 - 10 = -10$。交点 $(0, -10)$。
  \end{solutionblock}
\end{question}



\begin{question}[points=12]
  直线 $y = x - 2$ 与 $y = -3x + 10$ 的交点坐标为
  \begin{choices}
    \item $(2, 0)$
    \item $(3, 1)$
    \item $(1, -1)$
    \item $(3, 4)$
  \end{choices}
  \paren[B]
  \begin{solutionblock}
    令 $x-2=-3x+10$，$4x=12$，$x=3$。$y=3-2=1$。交点 $(3, 1)$。
  \end{solutionblock}
\end{question}


\section*{二、填空题（每题 12 分，共 12 分）}



\begin{question}[points=12]
  直线 $y = 2kx + 6$ 经过点 $(1, 4)$，则 $k$ = \rule{3cm}{0.4pt}。
  \fillin
  \paren[$-1$]
  \begin{solutionblock}
    代入 $(1, 4)$：$4=2k+6$，$2k=-2$，$k=-1$。
  \end{solutionblock}
\end{question}

\begin{problem}[points=20]
  直线 $y = \dfrac{1}{3}x - 4$ 与 $x$ 轴、$y$ 轴分别交于点 $A$、$B$。

求 $A$、$B$ 的坐标。
  \answerarea[50mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$A(12, 0)$，$B(0, -4)$\par\medskip
    \solstep{求 $B$（$y$ 轴交点）}{令 $x=0$：$y = -4$，$B(0, -4)$。}
    \solstep{求 $A$（$x$ 轴交点）}{令 $y=0$：$0 = \frac{1}{3}x - 4$，$\frac{1}{3}x = 4$，$x = 12$，$A(12, 0)$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=20]
  直线 $l_1$：$y = 2x + 2$ 与直线 $l_2$：$y = -x + 8$。

求 $l_1$ 与 $l_2$ 的交点坐标。
  \answerarea[50mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$(2, 6)$\par\medskip
    \solstep{联立}{$2x + 2 = -x + 8$}
    \solstep{解 $x$}{$3x = 6$，$x = 2$}
    \solstep{回代}{代入 $l_1$：$y = 2 \times 2 + 2 = 6$。交点 $(2, 6)$。}
  \end{solutionblock}
\end{problem}
