\section*{一、找出 A 字型和 8 字型}

\needspace{8\baselineskip}

\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    观察下图，填写：（1）两条平行线；（2）平行线外侧的点；（3）过该点且同时与 $l_1,l_2$ 相交的两条直线；（4）相似三角形；（5）模型名称。

  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{第 1 题图}
  \input{\detokenize{build/diagram/jobs/recognition-teacher-q1-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[34mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$l_1\parallel l_2$；点 $A$；直线 $AC$、$AD$；$\triangle ACD\sim\triangle AEF$；A 字型。\par\medskip
    \solstep{找平行线}{$l_1\parallel l_2$。}
    \solstep{找外点和两条截线}{外侧点为 $A$，过 $A$ 的两条截线为 $AC$、$AD$。}
    \solstep{写相似模型}{$\triangle ACD\sim\triangle AEF$，且 A 在平行线外侧，所以是 A 字型。}
  \end{solutionblock}
\end{problem}
