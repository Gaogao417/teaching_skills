\section*{一、平行线对应边比例（A 字型 2 题，8 字型 2 题）}

\needspace{8\baselineskip}

\begin{problem}[points=8]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    如图，$l_1\parallel l_2$，点 $A$ 在两条平行线外侧，直线 $AC$、$AD$ 分别交 $l_1$、$l_2$ 于 $C,E$ 和 $D,F$。已知 $AC:CE=2:3$，$AD=8$，求 $DF$。

  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{第 1 题图}
  \input{\detokenize{build/diagram/jobs/practice-ratio-teacher-q1-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[30mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$DF=12$。\par\medskip
    \solstep{先看已知比的对象}{$AC:CE$ 是同一条截线上的分段比；图形是 A 字型。}
    \solstep{写成对应份数}{$AD:DF=AC:CE=2:3$。}
    \solstep{用份数求解}{$AD=8$ 是 2 份，每份 $8\div2=4$，所以 $DF=3\times4=12$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=8]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    如图，$l_1\parallel l_2$，点 $A$ 在两条平行线外侧，直线 $AC$、$AD$ 分别交 $l_1$、$l_2$ 于 $C,E$ 和 $D,F$。已知 $AC:AE=3:5$，$CD=9$，求 $EF$。

  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{第 2 题图}
  \input{\detokenize{build/diagram/jobs/practice-ratio-teacher-q2-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[30mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$EF=15$。\par\medskip
    \solstep{先看已知比的对象}{$AC$、$AE$ 分属两个相似三角形，是对应边比。}
    \solstep{配对对应边}{由 $\triangle ACD\sim\triangle AEF$，得 $CD:EF=AC:AE=3:5$。}
    \solstep{用份数求解}{$CD=9$ 是 3 份，每份 3，所以 $EF=5\times3=15$。}
  \end{solutionblock}
\end{problem}
