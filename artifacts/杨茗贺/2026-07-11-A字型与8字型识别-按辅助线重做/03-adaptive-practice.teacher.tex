\needspace{8\baselineskip}

\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    观察图中的平行标记，按学生版的五步填写一个相似模型。

  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{第 1 题图}
  \input{\detokenize{build/diagram/jobs/identify-teacher-q1-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[26mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$DG\parallel BC$；模型点为 $A$；截线为 $AB,AC$；$\triangle ADG\sim\triangle ABC$；A 字型。\par\medskip
    \solstep{找平行线}{$DG\parallel BC$。}
    \solstep{抽取模型}{取点 $A$，两条截线为 $AB,AC$，得到 $\triangle ADG\sim\triangle ABC$。}
  \end{solutionblock}
\end{problem}
