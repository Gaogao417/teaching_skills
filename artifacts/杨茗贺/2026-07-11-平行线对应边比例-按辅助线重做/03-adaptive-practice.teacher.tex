\section*{平行辅助线图中的对应边比例}

\section*{一、先辨已知比，再用份数求解}

\needspace{8\baselineskip}

\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    如图，$DG\parallel BC$。已知 $AD:DB=2:1$，$AC=18$，求 $AG$。

  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{第 1 题图}
  \input{\detokenize{build/diagram/jobs/ratio-teacher-q1-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[24mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}分段比；$AG=12$。\par\medskip
    \solstep{判别已知比}{$AD:DB=2:1$ 是截线 $AB$ 上的分段比。}
    \solstep{合成整段}{$AD:AB=2:(2+1)=2:3$。}
    \solstep{对应求值}{$AG:AC=2:3$，所以 $AG=18\times\dfrac23=12$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    如图，$DG\parallel BC$。已知 $DG:BC=3:5$，$GC=8$，求 $AG$。

  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{第 2 题图}
  \input{\detokenize{build/diagram/jobs/ratio-teacher-q2-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[24mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}对应边比；$AG=12$。\par\medskip
    \solstep{判别已知比}{$DG:BC=3:5$ 是两相似三角形的对应边比。}
    \solstep{求剩余份数}{$AG:GC=3:(5-3)=3:2$。}
    \solstep{按份数求值}{$GC=8$ 对应 2 份，故 $AG=8\times\dfrac32=12$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    如图，$AB\parallel CF$。已知 $AE:EC=5:6$，$AB=15$，求 $CF$。

  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{第 3 题图}
  \input{\detokenize{build/diagram/jobs/ratio-teacher-q3-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[24mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}对应边比；$CF=18$。\par\medskip
    \solstep{判别已知比}{$AE:EC$ 是两个小三角形的对应边比。}
    \solstep{写对应比例}{由 $\triangle EAB\sim\triangle ECF$，$AB:CF=AE:EC=5:6$。}
    \solstep{求值}{$CF=15\times\dfrac65=18$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    如图，$DB\parallel CF$。已知 $CP:PD=4:1$，$DB=6$，求 $CF$。

  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{第 4 题图}
  \input{\detokenize{build/diagram/jobs/ratio-teacher-q4-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[24mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}对应边比；$CF=24$。\par\medskip
    \solstep{对齐比例顺序}{$CP:PD=4:1$，所以 $PD:PC=1:4$。}
    \solstep{写对应比例}{由 $\triangle PDB\sim\triangle PCF$，$DB:CF=PD:PC=1:4$。}
    \solstep{求值}{$CF=6\times4=24$。}
  \end{solutionblock}
\end{problem}
