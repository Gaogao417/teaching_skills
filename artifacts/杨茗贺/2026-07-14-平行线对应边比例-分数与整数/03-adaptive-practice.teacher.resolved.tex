\section*{三、平行线比例：已知三条线求第四条线（10 道）}

\needspace{8\baselineskip}

\begin{problem}[points=10]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-60mm-6mm\relax}
    \vspace{0pt}
    如图，点序分别为 $P-A-C$、$P-B-D$，且 $AB\parallel CD$。
已知 $PA=\frac{5}{12}$，$PC=\frac{25}{12}$，$CD=10$，求 $AB$。


  \end{minipage}\hfill
  \begin{diagramcoltikz}{60mm}{本题图}
  \input{\detokenize{build/diagram/jobs/parallel-fraction-q1-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[38mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$AB=2$。\par\medskip
    \solstep{写比例}{$\frac{AB}{10}=\frac{5/12}{25/12}=\frac15$。}
    \solstep{求第四条线}{$AB=10\times\frac15=2$。}
  \end{solutionblock}
\end{problem}
