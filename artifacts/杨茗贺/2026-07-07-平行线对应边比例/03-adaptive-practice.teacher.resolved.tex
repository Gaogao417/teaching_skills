\begin{problem}[points=8]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-56mm-6mm\relax}
    \vspace{0pt}
    如图，在 $\triangle ACE$ 中，点 $B$ 在线段 $AC$ 上，点 $D$ 在线段 $AE$ 上，且 $BD\parallel CE$。已知 $AD=4$，$AB=3$，$BC=6$，求 $DE$。

    \begin{hintbox}
\textbf{提示：}要求的边是 $DE$，这条边的对应边是 $BC$；先想 $DE:AD=BC:AB$。\end{hintbox}

  \end{minipage}\hfill
  \begin{diagramcoltikz}{56mm}{第 1 题图：$BD\parallel CE$}
  \input{\detokenize{build/diagram/jobs/practice-teacher-q1-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[30mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$DE=8$。\par\medskip
    \solstep{先定目标比}{要求 $DE$，已知 $AD=4$，先想 $DE:AD$。}
    \solstep{改成已知比}{由 $BD\parallel CE$，对应差段成比例，$DE:AD=BC:AB=6:3=2:1$。}
    \solstep{乘已知量}{$DE=AD\times2=4\times2=8$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=8]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-56mm-6mm\relax}
    \vspace{0pt}
    如图，在 $\triangle ACE$ 中，点 $B$ 在线段 $AC$ 上，点 $D$ 在线段 $AE$ 上，且 $BD\parallel CE$。已知 $AB=5$，$BC=3$，$BD=10$，求 $CE$。

    \begin{hintbox}
\textbf{提示：}要求的边是 $CE$，这条边的对应边是 $BD$；先把 $AB$ 和 $AC$ 的比例写出来。\end{hintbox}

  \end{minipage}\hfill
  \begin{diagramcoltikz}{56mm}{第 2 题图：$BD\parallel CE$}
  \input{\detokenize{build/diagram/jobs/practice-teacher-q2-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[30mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$CE=16$。\par\medskip
    \solstep{求整段}{$AC=AB+BC=5+3=8$。}
    \solstep{写平行线段比}{由 $BD\parallel CE$，$BD:CE=AB:AC=5:8$。}
    \solstep{乘已知量}{$CE=BD\times\dfrac{8}{5}=10\times\dfrac85=16$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=8]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-56mm-6mm\relax}
    \vspace{0pt}
    如图，在 $\triangle ACE$ 中，点 $B$ 在线段 $AC$ 上，点 $D$ 在线段 $AE$ 上，且 $BD\parallel CE$。已知 $AD=6$，$DE=4$，$CE=15$，求 $BD$。

    \begin{hintbox}
\textbf{提示：}要求的边是 $BD$，这条边的对应边是 $CE$；先把 $AD$ 和 $AE$ 的比例写出来。\end{hintbox}

  \end{minipage}\hfill
  \begin{diagramcoltikz}{56mm}{第 3 题图：$BD\parallel CE$}
  \input{\detokenize{build/diagram/jobs/practice-teacher-q3-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[30mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$BD=9$。\par\medskip
    \solstep{求整段}{$AE=AD+DE=6+4=10$。}
    \solstep{写平行线段比}{由 $BD\parallel CE$，$BD:CE=AD:AE=6:10=3:5$。}
    \solstep{乘已知量}{$BD=CE\times\dfrac35=15\times\dfrac35=9$。}
  \end{solutionblock}
\end{problem}

\begin{problem}[points=8]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-56mm-6mm\relax}
    \vspace{0pt}
    如图，在 $\triangle ACE$ 中，点 $B$ 在线段 $AC$ 上，点 $D$ 在线段 $AE$ 上，且 $BD\parallel CE$。已知 $AB=4$，$AC=10$，$DE=9$，求 $AD$。

    
  \end{minipage}\hfill
  \begin{diagramcoltikz}{56mm}{第 4 题图：$BD\parallel CE$}
  \input{\detokenize{build/diagram/jobs/practice-teacher-q4-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[32mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$AD=6$。\par\medskip
    \solstep{转成右侧比例}{$AB:AC=4:10=2:5$，所以 $AD:AE=2:5$。}
    \solstep{看差段}{$AE$ 比 $AD$ 多 $3$ 份，因此 $DE:AD=3:2$。}
    \solstep{乘已知量}{$AD=DE\times\dfrac23=9\times\dfrac23=6$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=8]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-56mm-6mm\relax}
    \vspace{0pt}
    如图，在 $\triangle ACE$ 中，点 $B$ 在线段 $AC$ 上，点 $D$ 在线段 $AE$ 上，且 $BD\parallel CE$。已知 $AD=5$，$AE=8$，$BD=6$，求 $CE$，并求 $DE:AD$。

    
  \end{minipage}\hfill
  \begin{diagramcoltikz}{56mm}{第 5 题图：$BD\parallel CE$}
  \input{\detokenize{build/diagram/jobs/practice-teacher-q5-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[34mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$CE=\dfrac{48}{5}$，$DE:AD=3:5$。\par\medskip
    \solstep{用平行线段比求 CE}{$BD:CE=AD:AE=5:8$，所以 $CE=BD\times\dfrac85=6\times\dfrac85=\dfrac{48}{5}$。}
    \solstep{求差段比例}{$DE=AE-AD=8-5=3$，所以 $DE:AD=3:5$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=8]
  \noindent
  \begin{minipage}[t]{\dimexpr\linewidth-56mm-6mm\relax}
    \vspace{0pt}
    如图，在 $\triangle ACE$ 中，点 $B$ 在线段 $AC$ 上，点 $D$ 在线段 $AE$ 上，且 $BD\parallel CE$。已知 $BD:CE=3:7$，$AD=6$，$BC=8$，求 $AB$ 和 $DE$。

    
  \end{minipage}\hfill
  \begin{diagramcoltikz}{56mm}{第 6 题图：$BD\parallel CE$}
  \input{\detokenize{build/diagram/jobs/practice-teacher-q6-prompt/rendered/prompt.fragment.tex}}
  \end{diagramcoltikz}
  \par\medskip
  \answerarea[36mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$AB=6$，$DE=8$。\par\medskip
    \solstep{把平行线段比改成两侧比例}{$BD:CE=3:7$，所以 $AB:AC=3:7$，$AD:AE=3:7$。}
    \solstep{求 AB}{左侧差段 $BC$ 是 $7-3=4$ 份，$BC=8$，每份为 $2$，所以 $AB=3\times2=6$。}
    \solstep{求 DE}{右侧 $AD$ 是 $3$ 份，$AD=6$，每份为 $2$。$DE$ 是 $4$ 份，所以 $DE=4\times2=8$。}
  \end{solutionblock}
\end{problem}
