\section*{双勾股求边长 · 专题练习}

\section*{一、选择题（每题 4 分，共 8 分）}


\begin{question}[points=4]
  在等腰三角形 $ABD$ 中，$AD=AB$。过 $A$ 作 $AH\perp BD$，垂足为 $H$。下列结论一定正确的是
  \begin{choices}
    \item $AH=BD$
    \item $BH=HD$
    \item $AB=BH$
    \item $AC=AD$
  \end{choices}
\end{question}



\begin{question}[points=4]
  设 $BC=x$，点序为 $B-C-H-D$，且 $BH=\dfrac94x$。则 $CH$ 等于
  \begin{choices}
    \item $\dfrac14x$
    \item $\dfrac54x$
    \item $\dfrac{13}4x$
    \item $\dfrac94x$
  \end{choices}
\end{question}


\section*{二、填空题（每题 5 分，共 10 分）}


\begin{question}[points=5]
  若 $BD=4BC$，过 $A$ 作 $AH\perp BD$ 后由 $AD=AB$ 得 $BH=\underline{\quad\quad}BC$。
\end{question}



\begin{question}[points=5]
  若 $AB=15$，$AC=13$，$BH=9$，$CH=5$，则公共高 $AH=\underline{\quad\quad}$。
\end{question}
