\begin{eduProblemBlock}[题目]
如图，在三角形 $ABC$ 中，点 $D$ 在射线 $BC$ 上，且 $C$ 在 $B,D$ 之间，$AD=AB$。
已知 $AB=15$，$AC=13$，且 $2BD=9BC$，求 $BC$ 的长。

\end{eduProblemBlock}





\needspace{10\baselineskip}
\begin{center}
  \includegraphics[width=0.58\linewidth]{\detokenize{diagram/rendered/diagram.png}}
  \par\vspace{1mm}{\small\color{edu-muted}先看 $B,C,H,D$ 共线，再看 $AH\perp BD$ 与 $AD=AB$。}
\end{center}








\begin{eduAuxItem}{思路导航}
过 $A$ 作 $AH\perp BD$$\;\to\;$由 $AD=AB$ 得 $H$ 是 $BD$ 中点$\;\to\;$设 $BC=x$，表示 $BH$ 和 $CH$$\;\to\;$在 $\triangle ABH$、$\triangle ACH$ 中两次勾股$\;\to\;$两式相减，消去 $AH^2$\end{eduAuxItem}




\begin{eduAuxItem}{核心思路}
看到 $AD=AB$，要把它放进等腰三角形 $ABD$ 里看。
过顶点 $A$ 作底边 $BD$ 的高 $AH$，这个高同时也是中线，所以 $BH=HD$。
接下来两个直角三角形共用同一条高 $AH$，两次勾股相减就能把 $AH^2$ 消掉。
\end{eduAuxItem}




\begin{eduSubQuestionItem}{(1)}
为什么要作 $AH\perp BD$？\end{eduSubQuestionItem}

\begin{eduSolutionBox}{思考引导}
  \begingroup
  \setlength{\parskip}{2.5mm}
$AD=AB$ 说明 $\triangle ABD$ 是等腰三角形。\par
等腰三角形里，顶点到底边的高还有什么身份？\par
  \endgroup
  \vspace{2.5mm}
  {\color{edu-blue}\bfseries\small 入口}\par\vspace{1.5mm}
  \begin{eduBulletList}
    \item 这一步把 $AD=AB$ 翻译成了可计算的线段关系。  \end{eduBulletList}
\end{eduSolutionBox}




\begin{eduSubQuestionItem}{(2)}
怎样把 $BH$、$CH$ 都用 $BC$ 表示？\end{eduSubQuestionItem}

\begin{eduSolutionBox}{思考引导}
  \begingroup
  \setlength{\parskip}{2.5mm}
设 $BC=x$，先由 $2BD=9BC$ 求 $BD$。\par
注意点序是 $B-C-D$，本题中 $H$ 在 $C,D$ 之间。\par
  \endgroup
  \vspace{2.5mm}
  {\color{edu-blue}\bfseries\small 停一下}\par\vspace{1.5mm}
  \begin{eduBulletList}
    \item 这一步最容易错在 $CH$：它不是 $\dfrac94x+x$，而是 $BH-BC$。  \end{eduBulletList}
\end{eduSolutionBox}




\begin{eduSubQuestionItem}{(3)}
怎样用两次勾股求出 $BC$？\end{eduSubQuestionItem}

\begin{eduSolutionBox}{思考引导}
  \begingroup
  \setlength{\parskip}{2.5mm}
$\triangle ABH$ 和 $\triangle ACH$ 都是直角三角形。\par
两个式子里都有 $AH^2$，相减可以把它消掉。\par
  \endgroup
  \vspace{2.5mm}
  {\color{edu-blue}\bfseries\small 答案}\par\vspace{1.5mm}
  \begin{eduBulletList}
    \item 因此 $BC=4$。  \end{eduBulletList}
\end{eduSolutionBox}




\begin{eduSubQuestionItem}{(4)}
怎样快速验算？\end{eduSubQuestionItem}

\begin{eduSolutionBox}{思考引导}
  \begingroup
  \setlength{\parskip}{2.5mm}
把 $BC=4$ 代回比例条件，看 $BH$ 和 $CH$ 是否干净。\par
如果出现两个经典直角三角形，说明计算很稳。\par
  \endgroup
  \vspace{2.5mm}
  {\color{edu-blue}\bfseries\small 结构}\par\vspace{1.5mm}
  \begin{eduBulletList}
    \item 这题实际藏了两个直角三角形：$9-12-15$ 和 $5-12-13$。  \end{eduBulletList}
\end{eduSolutionBox}




\begin{eduMistakeBox}
\eduSubTitle{只看到 $AD=AB$，没作底边高}$AD=AB$ 本身不能直接求 $BC$。它真正有用的地方是：作 $AH\perp BD$ 后，可以推出 $BH=HD$。
\end{eduMistakeBox}




\begin{eduMistakeBox}
\eduSubTitle{$CH$ 的方向写反}本题点序是 $B-C-H-D$，所以 $CH=BH-BC$。如果把它写成 $BH+BC$，后面勾股式会整体跑偏。
\end{eduMistakeBox}




\begin{eduAuxItem}{思路启发}
边讲边问：如果把 $2BD=9BC$ 改成 $BD=4BC$，你能先写出 $BH$ 和 $CH$ 吗？\end{eduAuxItem}




\begin{eduBottomGrid}
  \begin{eduSummaryLeft}
    \begin{eduSummaryBlock}{\eduIconCheck}{答案}
    \begin{eduPlainList}
      \item $BC=4$    \end{eduPlainList}
    \end{eduSummaryBlock}
  \end{eduSummaryLeft}
  \eduSummaryDivider
  \begin{eduSummaryRight}
    \begin{eduSummaryBlock}{\eduIconPin}{方法提醒}
    \begin{eduBulletList}
      \item 等腰三角形里，先作底边高，把等边关系转成中点关系。      \item 两个直角三角形共高时，优先考虑两次勾股相减。    \end{eduBulletList}
    \end{eduSummaryBlock}
  \end{eduSummaryRight}
\end{eduBottomGrid}
