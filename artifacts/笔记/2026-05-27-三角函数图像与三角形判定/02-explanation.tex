\begin{eduProblemBlock}[题目]
已知函数 $y=f(x)$，$f(x)=A\sin(\omega x+\varphi)$，其中 $A>0,\omega>0,|\varphi|<\dfrac{\pi}{2}$，其部分图像如图所示。
\begin{enumerate}[label=(\arabic*)]
  \item 求 $y=f(x)$ 的解析式；
  \item 在 $\triangle ABC$ 中，$a,b,c$ 分别是角 $A,B,C$ 的对边，若 $f(B)=1$，$a,b,c$ 成等差数列，判断 $\triangle ABC$ 的形状。
\end{enumerate}

\end{eduProblemBlock}





\needspace{10\baselineskip}
\begin{center}
  \includegraphics[width=0.58\linewidth]{\detokenize{diagram/jobs/orig-prompt/rendered/prompt.png}}
  \par\vspace{1mm}{\small\color{edu-muted}先读最大值、$f(0)$ 和峰值横坐标。}
\end{center}








\begin{eduAuxItem}{思路导航}
读图：最大值 $2$，$f(0)=1$，峰值在 $x=\dfrac{\pi}{6}$$\;\to\;$求参：先得 $A,\varphi$，再由峰值相位求 $\omega$$\;\to\;$定角：由 $f(B)=1$ 求 $B$$\;\to\;$判形：$2b=a+c$ 与余弦定理联立\end{eduAuxItem}




\begin{eduAuxItem}{核心思路}
这道题的难点不是某一步很长，而是每个条件都要翻译准确。图像条件翻译成三角函数参数，边成等差翻译成 $2b=a+c$，角 $B$ 再交给余弦定理。
\end{eduAuxItem}




\begin{eduSubQuestionItem}{(1)}
求 $y=f(x)$ 的解析式。\end{eduSubQuestionItem}

\begin{eduSolutionBox}{先读图}
  \begingroup
  \setlength{\parskip}{2.5mm}
最大值是 $2$，且 $A>0$，所以 $A=2$。\par
曲线经过 $(0,1)$，所以 $f(0)=1$。\par
图中峰值对应 $x=\dfrac{\pi}{6}$。\par
  \endgroup
  \vspace{2.5mm}
  {\color{edu-blue}\bfseries\small 注意}\par\vspace{1.5mm}
  \begin{eduBulletList}
    \item $\omega$ 的多解不能全保留，要用图像的周期信息筛掉。  \end{eduBulletList}
\end{eduSolutionBox}




\needspace{10\baselineskip}
\begin{center}
  \includegraphics[width=0.58\linewidth]{\detokenize{diagram/jobs/orig-solution/rendered/prompt.png}}
  \par\vspace{1mm}{\small\color{edu-muted}把 $f(0)$ 与峰值点分别对应到相位方程。}
\end{center}




\begin{eduSubQuestionItem}{(2)}
若 $f(B)=1$，$a,b,c$ 成等差数列，判断 $\triangle ABC$ 的形状。\end{eduSubQuestionItem}

\begin{eduSolutionBox}{先求角，再看边}
  \begingroup
  \setlength{\parskip}{2.5mm}
$f(B)=1$ 先给出关于 $B$ 的三角方程。\par
$B$ 是三角形内角，所以 $B\in(0,\pi)$。\par
$a,b,c$ 成等差数列要写成 $2b=a+c$。\par
  \endgroup
  \vspace{2.5mm}
  {\color{edu-blue}\bfseries\small 别跳步}\par\vspace{1.5mm}
  \begin{eduBulletList}
    \item 不能因为 $B=60^\circ$ 就直接说等边，还需要证明三边相等。  \end{eduBulletList}
\end{eduSolutionBox}




\needspace{10\baselineskip}
\begin{center}
  \includegraphics[width=0.42\linewidth]{\detokenize{diagram/jobs/tri-solution/rendered/prompt.png}}
  \par\vspace{1mm}{\small\color{edu-muted}对照角 $B$、对边 $b$ 和等差关系。}
\end{center}




\begin{eduMistakeBox}
\eduSubTitle{易错提醒}余弦定理中，角 $B$ 的对边是 $b$，所以应写
$b^2=a^2+c^2-2ac\cos B$。如果把 $a$ 或 $c$ 放到左边，后面的判形就会错。
\end{eduMistakeBox}
