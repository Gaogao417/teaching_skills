\section*{矩形存在性综合作业}

\section*{一、图象中的关键点（每题 8 分，共 24 分）}


\begin{question}[points=8]
  点 $P$ 在第一象限，过 $P$ 分别作两坐标轴的垂线，所得矩形面积为 $8$。若 $P$ 又在直线 $y=x+2$ 上，则 $P$ 的坐标与该矩形面积分别为（\quad）。

  \begin{choices}
    \item $P(2,4)$，面积 $8$
    \item $P(4,2)$，面积 $8$
    \item $P(2,4)$，面积 $4$
    \item $P(-4,-2)$，面积 $8$
  \end{choices}
\end{question}



\begin{question}[points=8]
  点 $P$ 在第一象限，过 $P$ 分别作两坐标轴的垂线，所得矩形面积为 $12$。若 $P$ 还在直线 $y=x+1$ 上，以 $A(0,0)$，$B(3,0)$，$C=P$ 为相邻顶点作矩形 $ABCD$，则 $D=$ \underline{\hspace{2cm}}。

\end{question}



\begin{question}[points=8]
  已知 $A(0,0)$，$B(6,0)$，点 $C$ 在直线 $y=x$ 上。若以 $A,B,C,D$ 为四个顶点可以组成一个面积为 $18$ 的矩形，且点 $C$ 在反比例函数 $y=\dfrac{k}{x}$ 上，则 $k=$ \underline{\hspace{2cm}}。

\end{question}


\section*{二、函数条件下的矩形与菱形（每题 12 分，共 36 分）}

\needspace{8\baselineskip}

\begin{problem}[points=12]
  点 $P$ 在反比例函数 $y=\dfrac{k}{x}$ 的第一象限图象上。过 $P$ 分别作两坐标轴的垂线，所得矩形面积为 $20$，且 $x_P=4$。设 $A(0,0)$，$B(4,0)$，$C=P$，点 $D$ 在 $y$ 轴上，使四边形 $ABCD$ 为矩形。求 $k$、直线 $OP$ 的解析式以及点 $C,D$。

  \answerarea[55mm]
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=12]
  在直线 $y=x+2$ 上取点 $C$，使 $A(0,0)$，$B(2,0)$，$C,D$ 能构成矩形。若还要求点 $C$ 所在的位置满足 $x_C+2\ge\dfrac{8}{x_C}$，求所有可能的 $C,D$。

  \answerarea[65mm]
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=12]
  已知 $A(0,0)$，$B(4,0)$，点 $C$ 在直线 $y=x$ 上。若四边形 $ABCD$ 是菱形，求点 $C,D$；并判断所得菱形是否也是矩形。

  \answerarea[55mm]
\end{problem}


\section*{三、综合应用（每题 10 分，共 40 分）}

\needspace{8\baselineskip}

\begin{problem}[points=10]
  已知 $A(0,0)$，$B(8,0)$。矩形 $ABCD$ 的面积为 $16$，点 $C$ 在第一象限。若点 $C$ 同时在直线 $y=mx$ 与反比例函数 $y=\dfrac{k}{x}$ 上，求 $C,D,m,k$。

  \answerarea[58mm]
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  已知 $A(0,0)$，$B(4,0)$，点 $C$ 在直线 $y=2x$ 上。是否存在点 $C,D$，使四边形 $ABCD$ 既是矩形又是菱形？请说明理由。

  \answerarea[52mm]
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  点 $P$ 在第一象限，过 $P$ 分别作两坐标轴的垂线，所得矩形面积为 $18$。若点 $P$ 又在直线 $y=x+3$ 上，过 $O(0,0)$，$P$ 作直线。现有 $A(0,0)$，$B(6,0)$，在直线 $OP$ 上取点 $E$，使四边形 $ABEF$ 为菱形。求点 $P$、直线 $OP$ 的解析式以及点 $E,F$。

  \answerarea[75mm]
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  点 $P$ 在反比例函数 $y=\dfrac{k}{x}$ 的第一象限图象上。过 $P$ 分别作两坐标轴的垂线，所得矩形面积为 $12$。若点 $P$ 又在直线 $y=x+1$ 上，设 $A(0,0)$，$B(x_P,0)$，$C=P$，四边形 $ABCD$ 为矩形。若直线 $OC$ 的解析式为 $y=mx$，求 $k,P,D,m$。

  \answerarea[75mm]
\end{problem}
