\eduSectionTitle{模型一：固定顺序矩形与相邻边垂直}


\begin{eduSolutionBox}{核心结论：先定直角，再求第四点}
  \begingroup
  \setlength{\parskip}{2.5mm}
\textbf{固定顺序：} 题目写“四边形 $ABCD$ 是矩形”，相邻边是 $AB$ 与 $BC$，所以先用 $AB\perp BC$。\par
\textbf{第四点：} 求出 $C$ 后，矩形也是平行四边形，所以 $D=A+C-B$。\par
\textbf{坐标工具：} 水平边看横坐标，斜边用向量点积 $\overrightarrow{AB}\cdot\overrightarrow{BC}=0$。\par
  \endgroup
\end{eduSolutionBox}




\begin{eduProblemBlock}[例题 1]
已知 $A(0,0)$，$B(4,0)$，点 $C$ 在反比例函数 $y=\dfrac{12}{x}$ 的图像上，且 $x_C>0$。
若四边形 $ABCD$ 是矩形，求点 $C,D$ 的坐标。

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
固定顺序找 $\angle B$$\;\to\;$代入轨迹求 $C$$\;\to\;$用平行四边形公式求 $D$\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 1}
固定顺序矩形怎样快速确定动点？\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{先圈题干顺序}{看到“四边形 $ABCD$”，先把 $AB$ 和 $BC$ 当相邻边。}
        \eduSideMistake{不要先猜 $D$}{矩形条件先限制 $C$，$D$ 通常最后用坐标和求。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{固定顺序找 $\angle B$}{$AB$ 水平，且矩形 $ABCD$ 中 $BC\perp AB$，所以 $x_C=x_B=4$。}
      \eduExplainStep{2}{代入轨迹求 $C$}{代入 $y=\dfrac{12}{x}$，得 $C(4,3)$。}
      \eduExplainStep{3}{用平行四边形公式求 $D$}{$D=A+C-B=(0+4-4,0+3-0)=(0,3)$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduSummaryBlock}{\eduIconPin}{方法提醒}
\begin{eduBulletList}
  \item 水平边：垂线是竖直线，直接确定横坐标。  \item 斜边：写两个相邻边向量，点积为 $0$。  \item 求完动点后，第四点用 $D=A+C-B$。\end{eduBulletList}
\end{eduSummaryBlock}

\eduSectionTitle{模型二：四点构成矩形与三类直角}


\begin{eduSolutionBox}{核心结论：构成题必须分类}
  \begingroup
  \setlength{\parskip}{2.5mm}
\textbf{无固定顺序：} 题目只说 $A,B,C,D$ 构成矩形，不能默认顺序是 $ABCD$。\par
\textbf{三类直角：} 依次检查 $\angle A=90^\circ$、$\angle B=90^\circ$、$\angle C=90^\circ$。\par
\textbf{对应第四点：} 直角在 $A$ 时 $D=B+C-A$；直角在 $B$ 时 $D=A+C-B$；直角在 $C$ 时 $D=A+B-C$。\par
  \endgroup
\end{eduSolutionBox}




\begin{eduProblemBlock}[例题 2]
已知 $A(0,0)$，$B(6,0)$，点 $C$ 在直线 $y=2x$ 上。
若 $A,B,C,D$ 构成矩形，求所有可能的点 $C,D$。

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
设动点$\;\to\;$直角在 $A$$\;\to\;$直角在 $B$$\;\to\;$直角在 $C$\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 2}
构成矩形时怎样保证不漏解？\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{分类不是装饰}{每一种直角位置都对应不同的第四点公式。}
        \eduSideMistake{退化要舍去}{$C=A$ 或 $C=B$ 时不能构成四边形。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{设动点}{设 $C(t,2t)$。}
      \eduExplainStep{2}{直角在 $A$}{$AC\perp AB$ 时 $t=0$，此时 $C=A$，退化，舍去。}
      \eduExplainStep{3}{直角在 $B$}{$BC\perp AB$ 时 $t=6$，所以 $C(6,12)$，$D=A+C-B=(0,12)$。}
      \eduExplainStep{4}{直角在 $C$}{列 $(A-C)\cdot(B-C)=0$，得 $(-t,-2t)\cdot(6-t,-2t)=0$，解得 $t=0$ 或 $t=\dfrac65$。舍去 $t=0$，得 $C\left(\dfrac65,\dfrac{12}{5}\right)$，$D=A+B-C=\left(\dfrac{24}{5},-\dfrac{12}{5}\right)$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduSummaryBlock}{\eduIconPin}{方法提醒}
\begin{eduBulletList}
  \item 先写三行：直角在 $A$、直角在 $B$、直角在 $C$。  \item 每行求出 $C$ 后，立刻匹配对应的 $D$ 公式。  \item 最后检查点是否重合、是否满足原轨迹。\end{eduBulletList}
\end{eduSummaryBlock}

\eduSectionTitle{模型三：轨迹包装、参数反求与范围检查}


\begin{eduSolutionBox}{核心结论：参数题也先回到矩形条件}
  \begingroup
  \setlength{\parskip}{2.5mm}
动点在函数图像上，先写成 $C(t,f(t))$ 或先用垂直关系确定某个坐标。\par
出现面积时，注意高度或边长可能要取绝对值。\par
出现参数时，先由矩形关系确定结构，再用面积、轨迹或参数条件筛选。\par
  \endgroup
\end{eduSolutionBox}




\begin{eduProblemBlock}[例题 3]
已知 $A(0,0)$，$B(6,0)$，点 $C$ 在直线 $y=kx$ 上。
若四边形 $ABCD$ 是矩形，且矩形面积为 $36$，求 $k$ 的值以及点 $C,D$ 的坐标。

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
先由直角确定 $C$ 的形式$\;\to\;$用面积反求参数$\;\to\;$写出两种情况\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 3}
参数题怎样避免漏掉正负情况？\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{先结构，后参数}{不要先猜 $k$，先用矩形确定 $C$ 的横坐标。}
        \eduSideMistake{面积要看绝对值}{点 $C$ 可以在 $x$ 轴上方，也可以在下方。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{先由直角确定 $C$ 的形式}{固定顺序下 $BC\perp AB$，所以 $x_C=6$。又 $C$ 在 $y=kx$ 上，故 $C(6,6k)$。}
      \eduExplainStep{2}{用面积反求参数}{矩形面积为 $6\cdot |6k|=36$，所以 $|k|=1$。}
      \eduExplainStep{3}{写出两种情况}{$k=1$ 时 $C(6,6),D(0,6)$；$k=-1$ 时 $C(6,-6),D(0,-6)$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduMistakeBox}
\eduSubTitle{总易错提醒}做矩形存在性题，先问自己：题目是否固定了顶点顺序？如果固定，用相邻边垂直；如果只说“构成”，先分类。最后一定检查退化、定义域和附加条件。
\end{eduMistakeBox}
