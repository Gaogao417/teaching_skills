\section*{梯形中的中点连线专项练习}

\section*{一、远迁移练习}

\needspace{8\baselineskip}

\begin{problem}[points=10]
  如图，在梯形 $ABCD$ 中，$AB\parallel CD$，$AB=7$，$CD=19$。
点 $E,F$ 分别是对角线 $AC,BD$ 的中点。求 $EF$ 的长。

  \answerareawithdiagramcol{38mm}{build/diagram/jobs/practice-teacher-trap1-prompt/rendered/prompt.png}{0.32\linewidth}{第1题图}
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$EF=6$。\par\medskip
    \solstep{补辅助中点}{取 $M$ 为 $AD$ 的中点，使 $M,E$ 和 $M,F$ 分别进入两个三角形的中位线。}
    \solstep{两次中位线}{在 $\triangle ACD$ 中，$ME=\frac12CD=9.5$；在 $\triangle ABD$ 中，$MF=\frac12AB=3.5$。}
    \solstep{比较两段}{$AB\parallel CD$，所以 $ME$ 与 $MF$ 同向共线，$EF=9.5-3.5=6$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  如图，在梯形 $ABCD$ 中，$AB\parallel CD$，$AB=8$。
点 $E,F$ 分别是对角线 $AC,BD$ 的中点，且 $EF=5$。求 $CD$ 的长。

  \answerareawithdiagramcol{38mm}{build/diagram/jobs/practice-teacher-trap2-prompt/rendered/prompt.png}{0.32\linewidth}{第2题图}
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$CD=18$。\par\medskip
    \solstep{写出结构关系}{对角线中点连线等于两底差的一半：$EF=\frac12(CD-AB)$。}
    \solstep{代入反求}{$5=\frac12(CD-8)$，所以 $CD-8=10$，$CD=18$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  如图，两条直线 $l_1\parallel l_2$。点 $A,B$ 在 $l_1$ 上，$AB=10$；
点 $C,D$ 在 $l_2$ 上，$CD=24$。连接 $AC$ 与 $BD$，点 $M,N$
分别是 $AC,BD$ 的中点。求 $MN$ 的长，并说明它和梯形题有什么共同结构。

  \answerareawithdiagramcol{42mm}{build/diagram/jobs/practice-teacher-parallel1-prompt/rendered/prompt.png}{0.32\linewidth}{第3题图}
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$MN=7$。共同结构：两条平行线之间有两条交叉截线，连接截线中点，长度等于两条平行端点距离之差的一半。\par\medskip
    \solstep{识别包装}{虽然题目没有说梯形，但 $AB\parallel CD$，连接 $AC,BD$ 后就是同一类结构。}
    \solstep{使用中点连线关系}{$M,N$ 是两条交叉截线的中点，因此 $MN=\frac12(CD-AB)$。}
    \solstep{代入计算}{$MN=\frac12(24-10)=7$。}
  \end{solutionblock}
\end{problem}
