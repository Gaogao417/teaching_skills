\title{一次函数概念辨析}
\subject{图像与解析式 · 讲解}
\maketitle

\section*{原题}


\begin{keyideabox}
\textbf{原题}\par  已知一次函数 $y=kx+b$ 的图像经过点 $(2,5)$ 和点 $(-1,-1)$，求 $k$ 和 $b$ 的值。
\end{keyideabox}



\section*{一、先把题目拆开}


\begin{routemap}
  \begin{enumerate}
    \item 已知：两个点的坐标 $(2,5)$ 和 $(-1,-1)$
    \item 要求：求 $k$ 和 $b$
    \item 关键词：一次函数、经过点、待定系数法
    \item 忽略：无
  \end{enumerate}
\end{routemap}



\section*{二、解题路线}


\begin{routemap}
  \begin{enumerate}
    \item 将两个点代入 $y=kx+b$，列方程组
    \item 解方程组，求出 $k$ 和 $b$
    \item 写出最终解析式并检验
  \end{enumerate}
\end{routemap}



\section*{三、关键想法}


\begin{keyideabox}
  一次函数有两个未知系数 $k$ 和 $b$，两个独立条件可以确定它们。将点的坐标代入，得到关于 $k, b$ 的二元一次方程组。
\end{keyideabox}



\section*{四、标准解法}


\begin{keyideabox}
  \textbf{代入列方程}
  将 $(2,5)$ 代入：$5 = 2k + b$。将 $(-1,-1)$ 代入：$-1 = -k + b$。
\end{keyideabox}




\begin{keyideabox}
  \textbf{解方程组}
  两式相减：$6 = 3k$，所以 $k = 2$。代回：$5 = 4 + b$，所以 $b = 1$。
\par\small 待定系数法的核心是：用已知条件确定未知系数。\end{keyideabox}




\begin{keyideabox}
  \textbf{写出答案}
  $k = 2, b = 1$，即 $y = 2x + 1$。
  \par\hspace{5mm}\small — 检验：$(2, 2\times2+1) = (2,5)$ ✓
  \par\hspace{5mm}\small — 检验：$(-1, 2\times(-1)+1) = (-1,-1)$ ✓
\end{keyideabox}



\section*{五、易错提醒}


\begin{mistakebox}
\textbf{代入时符号出错}\par  将 $(-1,-1)$ 代入 $y=kx+b$ 时，容易写成 $-1 = k + b$（漏掉负号）。正确应为 $-1 = -k + b$。
\end{mistakebox}




\begin{mistakebox}
\textbf{忘记检验}\par  求出 $k, b$ 后应代回原点检验，避免计算错误。
\end{mistakebox}



\section*{六、边讲边问}


\begin{hintbox}
  \textbf{提示 1：}如果有三个点都满足 $y=2x+1$，还需要全部代入吗？
\end{hintbox}




\begin{hintbox}
  \textbf{提示 1：}如果只给你一个点，能唯一确定 $k$ 和 $b$ 吗？
\end{hintbox}



\section*{七、一句话总结}


\begin{keyideabox}
  待定系数法 = 代入已知点 → 列方程组 → 解出系数 → 写出解析式。
\end{keyideabox}
