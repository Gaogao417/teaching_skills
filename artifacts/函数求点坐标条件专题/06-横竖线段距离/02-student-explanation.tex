\eduSectionTitle{模型六：横竖距离转成坐标差}


\begin{eduSolutionBox}{核心结论：先判断看哪个坐标}
  \begingroup
  \setlength{\parskip}{2.5mm}
水平距离只看横坐标差：$|x_1-x_2|$。\par
竖直距离只看纵坐标差：$|y_1-y_2|$。\par
点到 $x$ 轴的距离是 $|y|$；点到 $y$ 轴的距离是 $|x|$。\par
如果题目没有给方向、象限或范围，距离条件常常产生两个候选点。\par
  \endgroup
\end{eduSolutionBox}




\begin{eduProblemBlock}[例题 1]
点 $A(-2,3)$，点 $B$ 与点 $A$ 在同一条水平线上，且 $AB=6$。
若点 $B$ 在点 $A$ 的右侧，求点 $B$ 的坐标。

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
同水平先定纵坐标$\;\to\;$右侧距离定横坐标$\;\to\;$写出点坐标\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 1}
用水平距离和方向求点坐标。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{水平看横}{同一水平线说明纵坐标不变，距离只改变横坐标。}
        \eduSideMistake{方向决定加减}{右侧是横坐标增加，左侧才是横坐标减少。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{同水平先定纵坐标}{$B$ 与 $A$ 在同一条水平线上，所以 $B$ 的纵坐标也是 $3$。}
      \eduExplainStep{2}{右侧距离定横坐标}{$B$ 在 $A$ 的右侧，且 $AB=6$，所以横坐标为 $-2+6=4$。}
      \eduExplainStep{3}{写出点坐标}{因此 $B(4,3)$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduProblemBlock}[例题 2]
点 $P$ 在一次函数 $y=-2x+7$ 的图像上，且点 $P$ 到 $x$ 轴的距离为 $3$。
若点 $P$ 在第一象限，求点 $P$ 的坐标。

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
把到 $x$ 轴距离转成纵坐标$\;\to\;$代入函数求横坐标$\;\to\;$写点并验算\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 2}
用点到轴距离和函数轨迹求点。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{到 $x$ 轴看纵}{离 $x$ 轴有多远，是看点的纵坐标离 $0$ 有多远。}
        \eduSideMistake{绝对值要筛选}{$|y|=3$ 原本有 $y=3$ 或 $y=-3$，第一象限帮我们选 $y=3$。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{把到 $x$ 轴距离转成纵坐标}{点到 $x$ 轴距离为 $3$，得 $|y|=3$。第一象限中 $y>0$，所以 $y=3$。}
      \eduExplainStep{2}{代入函数求横坐标}{将 $y=3$ 代入 $y=-2x+7$，得 $3=-2x+7$，解得 $x=2$。}
      \eduExplainStep{3}{写点并验算}{所以 $P(2,3)$，它在第一象限且到 $x$ 轴距离为 $3$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduMistakeBox}
\eduSubTitle{易错提醒}看到“距离”不要马上套两点距离公式。先判断这段距离是不是水平或竖直：水平只看横坐标，竖直只看纵坐标。点到 $x$ 轴看 $|y|$，点到 $y$ 轴看 $|x|$。
\end{eduMistakeBox}
