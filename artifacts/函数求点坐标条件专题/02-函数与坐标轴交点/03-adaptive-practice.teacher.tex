\section*{函数求点坐标条件专题 2：函数与坐标轴交点}

\section*{练习题（每题 10 分，共 30 分）}

\needspace{8\baselineskip}

\begin{problem}[points=10]
  一次函数 $y=-2x+6$ 的图像分别与 $x$ 轴、$y$ 轴交于点 $A,B$。求点 $A,B$ 的坐标。

  \answerarea[38mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$A(3,0)$，$B(0,6)$。\par\medskip
    \solstep{求 $x$ 轴交点}{令 $y=0$，得 $0=-2x+6$，解得 $x=3$，所以 $A(3,0)$。}
    \solstep{求 $y$ 轴交点}{令 $x=0$，得 $y=6$，所以 $B(0,6)$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  一次函数 $y=kx+b$ 的图像与 $x$ 轴交于 $A(-2,0)$，与 $y$ 轴交于 $B(0,3)$。求这个一次函数的解析式。

  \answerarea[42mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$y=\dfrac32x+3$。\par\medskip
    \solstep{确定 $b$}{$B(0,3)$ 在图像上，代入 $y=kx+b$ 得 $b=3$。}
    \solstep{求 $k$}{$A(-2,0)$ 在图像上，代入 $y=kx+3$ 得 $0=-2k+3$，所以 $k=\dfrac32$。}
    \solstep{写解析式}{一次函数为 $y=\dfrac32x+3$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  已知直线 $l:y=-2x+6$ 与 $x$ 轴交于点 $A$，与 $y$ 轴交于点 $B$。点 $P$ 在直线 $y=x+1$ 上，且点 $P$ 的横坐标等于点 $A$ 的横坐标；点 $Q$ 在直线 $y=-x+8$ 上，且点 $Q$ 的纵坐标等于点 $B$ 的纵坐标。求点 $P,Q$ 的坐标。

  \answerarea[52mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$P(3,4)$，$Q(2,6)$。\par\medskip
    \solstep{求截距}{$l:y=-2x+6$ 与 $x$ 轴交于 $A(3,0)$，与 $y$ 轴交于 $B(0,6)$。}
    \solstep{求 $P$}{$x_P=3$，代入 $y=x+1$，得 $y_P=4$，所以 $P(3,4)$。}
    \solstep{求 $Q$}{$y_Q=6$，代入 $y=-x+8$，得 $6=-x+8$，解得 $x=2$，所以 $Q(2,6)$。}
  \end{solutionblock}
\end{problem}
