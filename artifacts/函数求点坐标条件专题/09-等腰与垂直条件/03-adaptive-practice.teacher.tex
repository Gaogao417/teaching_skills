\section*{函数求点坐标条件专题 09 · 等腰与垂直条件}

\section*{一、等腰与垂直条件怎么给坐标}

\needspace{8\baselineskip}

\begin{problem}[points=10]
  已知 $A(0,0)$，$B(5,0)$，点 $P$ 在直线 $y=3$ 上。
若 $\triangle ABP$ 是等腰三角形，求所有可能的点 $P$，并说明对应哪两条边相等。

  \answerarea[62mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$P\left(\dfrac52,3\right)$，此时 $PA=PB$；$P(4,3)$ 或 $P(-4,3)$，此时 $PA=AB$；$P(1,3)$ 或 $P(9,3)$，此时 $PB=AB$。\par\medskip
    \solstep{设点并写距离平方}{设 $P(t,3)$。则 $PA^2=t^2+9$，$PB^2=(t-5)^2+9$，$AB^2=25$。}
    \solstep{第一类：$PA=PB$}{$t^2+9=(t-5)^2+9$，解得 $t=\dfrac52$，所以 $P\left(\dfrac52,3\right)$。}
    \solstep{第二类：$PA=AB$}{$t^2+9=25$，解得 $t=4$ 或 $t=-4$，所以 $P(4,3)$ 或 $P(-4,3)$。}
    \solstep{第三类：$PB=AB$}{$(t-5)^2+9=25$，解得 $t=1$ 或 $t=9$，所以 $P(1,3)$ 或 $P(9,3)$。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  已知 $A(0,2)$，$B(4,0)$，点 $P$ 在一次函数 $y=x$ 的图像上。
若 $AP\perp BP$，求点 $P$ 的坐标。

  \answerarea[44mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$P(0,0)$ 或 $P(3,3)$\par\medskip
    \solstep{设函数图像上的点}{设 $P(t,t)$。}
    \solstep{写垂直条件}{$\overrightarrow{PA}=(-t,2-t)$，$\overrightarrow{PB}=(4-t,-t)$。垂直时数量积为 $0$。}
    \solstep{列方程并求解}{$(-t)(4-t)+(2-t)(-t)=0$，化简得 $2t^2-6t=0$，所以 $t=0$ 或 $t=3$。}
    \solstep{写坐标}{$P(0,0)$ 或 $P(3,3)$。其中 $P(0,0)$ 时一条线竖直、一条线水平，也满足垂直。}
  \end{solutionblock}
\end{problem}


\needspace{8\baselineskip}

\begin{problem}[points=10]
  已知直线 $l_1:y=-\dfrac43x+4$ 与直线 $l_2:y=-\dfrac34x+4$ 交于点 $A$，直线 $l_1$ 与 $x$ 轴交于点 $B$。
点 $P$ 在 $x$ 轴上，且 $AP\perp l_2$。求 $A,B,P$ 的坐标，并判断 $\triangle ABP$ 是否为等腰三角形。

  \answerarea[58mm]
  \setcounter{solstep}{0}
  \begin{solutionblock}
    \textbf{答案：}$A(0,4)$，$B(3,0)$，$P(-3,0)$；$AB=AP=5$，所以 $\triangle ABP$ 是等腰三角形。\par\medskip
    \solstep{求交点 $A$}{两条直线都有截距 $4$，且斜率不同，所以交点为 $A(0,4)$。}
    \solstep{求截距点 $B$}{令 $y=0$，$0=-\dfrac43x+4$，得 $x=3$，所以 $B(3,0)$。}
    \solstep{用垂直条件求 $P$}{$l_2$ 的斜率为 $-\dfrac34$，与它垂直的直线斜率为 $\dfrac43$。过 $A(0,4)$ 的直线为 $y=\dfrac43x+4$。令 $y=0$，得 $x=-3$，所以 $P(-3,0)$。}
    \solstep{判断等腰}{$AB=\sqrt{3^2+(-4)^2}=5$，$AP=\sqrt{(-3)^2+(-4)^2}=5$，所以 $AB=AP$，$\triangle ABP$ 是等腰三角形。}
  \end{solutionblock}
\end{problem}
