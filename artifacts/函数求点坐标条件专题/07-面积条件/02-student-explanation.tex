\eduSectionTitle{模型七：面积条件怎样变成坐标方程}


\begin{eduSolutionBox}{核心结论：面积给的是距离方程}
  \begingroup
  \setlength{\parskip}{2.5mm}
三角形面积公式：$S=\dfrac12\times \text{底}\times \text{高}$。\par
当底边在 $x$ 轴或水平线上时，高通常是纵坐标差的绝对值。\par
点在一次函数 $y=kx+b$ 上时，先设 $P(t,kt+b)$，再把面积条件写成关于 $t$ 的方程。\par
面积反求点常出现两个候选点；题目没有筛选条件时，要保留所有可能点。\par
  \endgroup
\end{eduSolutionBox}




\begin{eduProblemBlock}[例题 1]
已知 $A(1,0)$，$B(5,0)$，点 $P$ 在一次函数 $y=2x-1$ 的图像上。
若 $\triangle ABP$ 的面积为 $10$，求点 $P$ 的坐标。

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
设函数图像上的点$\;\to\;$用面积反求高$\;\to\;$解绝对值方程$\;\to\;$写出候选点\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 1}
由定面积求函数图像上的动点。\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{先看底边}{$A,B$ 都在 $x$ 轴上，所以把 $AB$ 当底边最省事。}
        \eduSideMistake{别漏绝对值}{点可以在 $x$ 轴上方，也可以在下方，距离要写成绝对值。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{设函数图像上的点}{设 $P(t,2t-1)$。}
      \eduExplainStep{2}{用面积反求高}{$AB=4$，点 $P$ 到 $x$ 轴的距离为 $|2t-1|$，所以 $\dfrac12\times4\times|2t-1|=10$。}
      \eduExplainStep{3}{解绝对值方程}{$|2t-1|=5$，所以 $t=3$ 或 $t=-2$。}
      \eduExplainStep{4}{写出候选点}{当 $t=3$ 时 $P(3,5)$；当 $t=-2$ 时 $P(-2,-5)$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduProblemBlock}[例题 2]
已知 $A(0,0)$，$B(6,0)$，$C(2,4)$，点 $P$ 在一次函数 $y=x-2$ 的图像上。
若 $\triangle ABP$ 与 $\triangle ABC$ 的面积相等，求所有可能的点 $P$。

\end{eduProblemBlock}





\begin{eduAuxItem}{思路导航}
先求参照面积$\;\to\;$设动点并列方程$\;\to\;$分两侧求点$\;\to\;$写出答案并验算\end{eduAuxItem}




\begin{eduSubQuestionItem}{例题 2}
等面积条件怎样转成两侧候选点？\end{eduSubQuestionItem}

\begin{eduDualExplain}
  \begin{eduExplainLeft}
    \eduColumnTitle{\eduIconBulb}{小贴士}
        \eduSideHint{等面积先等高}{同一底边下，面积相等就是到这条底边的距离相等。}
        \eduSideMistake{两侧都要看}{同样的距离可能在底边上方，也可能在底边下方。}
  \end{eduExplainLeft}

  \begin{eduExplainRight}
    \eduColumnTitle{\eduIconBook}{解答}
      \eduExplainStep{1}{先求参照面积}{$AB=6$，点 $C$ 到 $x$ 轴的距离为 $4$，所以 $S_{\triangle ABC}=\dfrac12\times6\times4=12$。}
      \eduExplainStep{2}{设动点并列方程}{设 $P(t,t-2)$，则 $\dfrac12\times6\times|t-2|=12$。}
      \eduExplainStep{3}{分两侧求点}{$|t-2|=4$，解得 $t=6$ 或 $t=-2$。}
      \eduExplainStep{4}{写出答案并验算}{所以 $P(6,4)$ 或 $P(-2,-4)$，两点到 $AB$ 所在直线的距离都为 $4$。}
  \end{eduExplainRight}
\end{eduDualExplain}




\begin{eduMistakeBox}
\eduSubTitle{易错提醒}面积条件最容易漏的是绝对值和筛选。先写距离方程，再看题目是否给了象限、线段范围或同侧条件。\end{eduMistakeBox}




\begin{eduSummaryBlock}{\eduIconPin}{方法提醒}
\begin{eduBulletList}
  \item 第一步找底边，优先选水平、竖直或已知长度的边。  \item 第二步把动点写成 $P(t,f(t))$。  \item 第三步列面积方程，保留所有候选点并逐一验算。\end{eduBulletList}
\end{eduSummaryBlock}
